router.get('/omr-uploads', async (req, res, next) => {
  try {
    const { _id: userId, districtId } = req.user
    const { assignmentId, groupId, archived, aborted } = req.query
    if (!assignmentId) {
      return res.boom.badRequest('assignmentId is required!')
    }
    const assignment = await new Assignment().getById(
      districtId,
      assignmentId,
      ['title', 'class', 'testId']
    )
    if (!assignment) {
      return res.boom.notFound('Invalid Assignment ID.')
    }
    const group = (assignment.class || []).find(
      (c) => `${c._id}` === `${groupId}`
    )
    if (!group) {
      return res.boom.notFound('Invalid Class ID.')
    }
 
    // region get `hasNonMcq` and `bubbleSheetVersion`
    const { testId } = assignment
    const test = await testModel._findByIdFromCache(testId)
    const testItemIds = test.itemGroups
      .flatMap((ig) => ig.items)
      .map((it) => it.itemId)
    const [testItems, bubbleSheetVersion] = await Promise.all([
      testItemModel._findByIdsFromCache(testItemIds),
      bubbleSheetQrLookupModel
        .getByAssignmentIdAndTestId(assignmentId, testId, {
          bubbleSheetVersion: 1,
        })
        .then((qrl) => qrl?.bubbleSheetVersion || config.bubbleSheetVersion),
    ])
    const hasNonMcq = testItems.some((ti) => {
      if (ti.data.questions.length !== 1 && !test.isDocBased) return true
      return ti.data.questions.some(
        (q) =>
          ![questionType.MULTIPLE_CHOICE, questionType.SECTION_LABEL].includes(
            // non-mcq !== [mcq, sectionLabel]
            q.type
          )
      )
    })
    // endregion
 
    const omrUploadSessions = await getOmrUploadSessions({
      userId,
      assignmentId,
      groupId,
      archived,
      aborted,
    })
    const result = {
      omrUploadSessions,
      assignmentTitle: assignment.title,
      classTitle: group.name,
      hasNonMcq,
      bubbleSheetVersion,
    }
    return generalHandle(res, { result })
  } catch (e) {
    next(e)
  }
})

router.post('/scan-omr', async (req, res, next) => {
  try {
    const { _id, role, districtId } = req.user
    const { assignmentId, groupId, sessionId, source } = req.body
    if ([assignmentId, sessionId].some((v) => !v)) {
      return res.boom.badRequest('assignmentId and sessionId are required!')
    }
    if (!source.uri) {
      return res.boom.badRequest('Source file is required!')
    }
    const isAllowedResp = await isBubbleSheetActionAllowed(
      assignmentId,
      sessionId
    )
    if (!isAllowedResp.isAllowed) {
      return res.boom.badRequest(
        'The upload has either been processed or cannot be processed at this time'
      )
    }
    // split scan controller
    const actionSummary = await splitScanBubbleSheets({
      uploadedBy: { _id, role, districtId },
      assignmentId,
      groupId,
      sessionId,
      source,
    })
    if (actionSummary.error) {
      const error = new Error(actionSummary.error)
      Object.assign(actionSummary, { error: error.message })
      ApiTraceLogger('error', 'BUBBLE_SCAN_ERROR', req, error, actionSummary)
    }
    return generalHandle(res, actionSummary)
  } catch (e) {
    next(e)
  }
})
 
router.post('/score-answers', async (req, res, next) => {
  try {
    /**
     * @type {{value:{assignmentId:number,groupId:number,sessionId:string,responses:{answers:string[],studentId:number,imageUri:string}[]}},error?:Error}
     */
    const body = joi.validate(req.body, webCamResponsesSchema)
    if (body.error) {
      ApiTraceLogger(
        'info',
        'VALIDATION_ERROR',
        req,
        new Error(body.error.message || 'JOI error')
      )
      return res.boom.badRequest(body.error.message)
    }
    let assignmentId
    let groupId
    let testId
    const {
      assignmentId: assignmentIdentifier,
      groupId: groupIdentifier,
      sessionId,
      responses,
    } = body.value
    if (isNaN(assignmentIdentifier) && isNaN(groupIdentifier)) {
      assignmentId = assignmentIdentifier
      groupId = groupIdentifier
    } else {
      const qrCodes = responses.map(
        (x) => `${assignmentIdentifier}.${groupIdentifier}.${x.studentId}`
      )
      const decoded = await decodeQrCodes(qrCodes)
      if (!decoded) {
        return res.boom.badRequest('Invalid Qr Codes')
      }
      assignmentId = `${decoded.assignmentId}`
      groupId = `${decoded.groupId}`
      testId = `${decoded.testId}`
 
      const { assignmentId: originalAssignmentId, groupId: originalGroupId } =
        (await omrUploadSessionModel.findOne(
          { _id: new ObjectId(sessionId) },
          { assignmentId: 1, groupId: 1 }
        )) || {}
 
      if (
        (originalAssignmentId &&
          `${originalAssignmentId}` != `${assignmentId}`) ||
        (originalGroupId && `${originalGroupId}` != `${groupId}`)
      ) {
        const message = 'The sheet does not belong to current assignment.'
        await omrUploadSessionModel.abortSession(sessionId, message)
        return generalHandle(res, { message }, 400)
      }
 
      const { testId: assignmentTestId } =
        (await new Assignment().getById(null, assignmentId, ['testId'])) || {}
 
      if (assignmentTestId && `${assignmentTestId}` != testId) {
        const message =
          'This test has been edited after the sheets were generated. Please enter the grades through the Express Grader or generate sheets again to take a new test.'
        await omrUploadSessionModel.abortSession(sessionId, message)
        return generalHandle(res, { message }, 400)
      }
 
      responses.forEach((res, index) => {
        res.studentId = `${decoded.studentIds[index]}`
      })
    }
    const questions = await getQuestionMetadataForOmrScanner({ assignmentId })
    const usernamesHash = await getUserNamesHash(
      responses.map((x) => x.studentId)
    )
    for (const response of responses) {
      const scoreAnswersPayload = {
        ...response,
        questions,
        assignmentId,
        groupId,
        sessionId,
        name: usernamesHash[`${response.studentId}`]?.name,
        action: assignmentBulkActions.WEB_SCAN_BUBBLE_SHEETS,
      }
 
      await sqs.send({
        QueueUrl: config.sqs.qtiImportUri,
        MessageBody: JSON.stringify(scoreAnswersPayload),
      })
    }
    return generalHandle(res, { result: 'Success' })
  } catch (e) {
    next(e)
  }
})
