import _ from 'lodash'
import _redshift, { getRedshiftInstance } from '../../services/redshift'
import { appendGroupQuery, getClassFilterQuery } from './commons'
import assignmentStatus from '../../const/assignmentStatus'
import { assignedBy as assignedByStatus } from '../../const/assignment'
 
export const getResponseFrequencyByQuestion = (
  termId: string,
  testId: string,
  districtId: string,
  schoolIds: string[],
  teacherIds: string[],
  subjects: string[],
  grades: string[],
  courseId: string,
  classIds: string[],
  groupIds: string[],
  assessmentTypes: string[],
  userRole: string,
  assignedBy: string,
  userId: string,
  race: string,
  gender: string,
  frlStatus: string,
  ellStatus: string,
  iepStatus: string,
  hispanicEthnicity: string
) => {
  const redshift = getRedshiftInstance() || _redshift
  const applyClassFilter = _.isEmpty(classIds) && _.isEmpty(groupIds)
 
  const classFilterQuery = getClassFilterQuery(
    districtId,
    termId,
    [courseId],
    grades,
    subjects,
    teacherIds,
    userRole,
    schoolIds,
    classIds
  )
 
  const assignmentFilterQuery = redshift({
    ta: 't_assignments',
  })
    .distinct('ta.assignment_id')
    .where({
      'ta.district_id': districtId,
      'ta.term_id': termId,
      'ta.report_key': testId,
    })
    .whereNot('ta.progress_status', assignmentStatus.NOT_OPEN)
  if (!_.isEmpty(assessmentTypes)) {
    assignmentFilterQuery.whereIn('ta.test_type', assessmentTypes)
  }
 
  const innerQuery = redshift({
    uqa: 't_user_test_standard_activity',
  })
    .select(
      'uqa.group_id',
      'uqa.assignment_id',
      'uqa.question_id',
      'uqa.is_correct',
      'uqa.is_skipped',
      'uqa.is_partial',
      'uqa.mhash',
      'uqa.score',
      'uqa.max_score',
      'uqa.usr_resp',
      'uqa.qtype'
    )
    .select(
      // in case of score missing in some standards that question belongs to
      // order by clause in partition will fetch the max of score amongst all standards
      redshift.raw(
        'row_number() over (partition by question_activity_id order by score DESC) row_num'
      )
    )
    .where({
      'uqa.district_id': districtId,
      'uqa.term_id': termId,
      'uqa.report_key': testId,
    })
    .whereIn('uqa.assignment_id', assignmentFilterQuery)
  if (assignedBy === assignedByStatus.ME) {
    innerQuery.where('uqa.assigned_by', userId)
    assignmentFilterQuery.where('ta.assigned_by', userId)
  }
  if (applyClassFilter) {
    innerQuery.whereIn('uqa.group_id', classFilterQuery)
  }
  if (!_.isEmpty(groupIds)) {
    appendGroupQuery(innerQuery, teacherIds, groupIds, classIds, 'uqa')
  } else if (!_.isEmpty(classIds)) {
    innerQuery.whereRaw(`uqa.group_id IN (${classIds.map((c) => `'${c}'`)})`)
  }
  if (!_.isEmpty(schoolIds)) {
    innerQuery.whereRaw(`uqa.school_id IN (${schoolIds.map((s) => `'${s}'`)})`)
  }
 
  if (
    _.some(
      [race, gender, frlStatus, ellStatus, iepStatus, hispanicEthnicity],
      'length'
    )
  ) {
    if (!_.isEmpty(race)) {
      innerQuery.where('usr.race', race)
    }
    if (!_.isEmpty(gender)) {
      innerQuery.where('usr.gender', gender)
    }
    if (!_.isEmpty(frlStatus)) {
      innerQuery.where('usr.frl_status', frlStatus)
    }
    if (!_.isEmpty(ellStatus)) {
      innerQuery.where('usr.ell_status', ellStatus)
    }
    if (!_.isEmpty(iepStatus)) {
      innerQuery.where('usr.iep_status', iepStatus)
    }
    if (!_.isEmpty(hispanicEthnicity)) {
      innerQuery.where('usr.hispanic_ethnicity', hispanicEthnicity)
    }
    innerQuery
      .joinRaw(`,"t_users" as "usr"`)
      .whereRaw('uqa.student_id = usr.user_id')
  }
 
  innerQuery.as('iq')
 
  return redshift()
    .select('question_id')
    .select(
      redshift.raw(
        'sum(CASE WHEN is_correct = 1 THEN 1 else 0 END) as corr_cnt'
      )
    )
    .select(
      redshift.raw(
        'sum(CASE WHEN is_correct != 1 and is_skipped != 1 and is_partial != 1 and score = 0 THEN 1 else 0 END) as incorr_cnt'
      )
    )
    .select(
      redshift.raw(
        'sum(CASE WHEN is_skipped = 1 THEN 1 else 0 END) as skip_cnt'
      )
    )
    .select(
      redshift.raw(
        'sum(CASE WHEN is_partial = 1 THEN 1 WHEN is_correct != 1 and is_skipped != 1 and score != 0 THEN 1 else 0 END) as part_cnt'
      )
    )
    .select(redshift.raw('sum(score) as total_score'))
    .select(redshift.raw('sum(max_score) as total_max_score'))
    .max('max_score as max_score')
    .select(
      redshift.raw(
        `listagg(distinct group_id, ',') within group (order by mhash) AS group_id`
      )
    )
    .select(
      redshift.raw(
        `listagg(distinct assignment_id, ',') within group (order by mhash) AS assignment_id`
      )
    )
    .select(
      redshift.raw(
        'listagg(distinct usr_resp) within group (order by mhash) as usr_resp'
      )
    )
    .select(
      redshift.raw(
        'listagg(distinct qtype) within group (order by mhash) as qtype'
      )
    )
    .from(innerQuery)
    .where('row_num', 1)
    .groupByRaw('question_id, mhash')
}
