export const getDistrictAverage = (
  testIds: string[],
  districtId: string,
  assignedBy: string,
  userId: string
) => {
  const redshift = getRedshiftInstance() || _redshift
  const rawDataQuery = redshift({
    uta: 't_user_test_activity',
  })
    .select('uta.test_id')
    .select(redshift.raw(`sum(score) / count(score) as district_avg`))
    .select(
      redshift.raw(
        `(sum(score) / nvl(nullif(sum(max_score),0),1))*100 as district_avg_perf`
      )
    )
    .where({
      'uta.progress_status': 1,
      'uta.district_id': districtId,
      'uta.archived': 0,
    })
  if (assignedBy === assignedByStatus.ME) {
    rawDataQuery.where('uta.assigned_by', userId)
  }
  if (!isEmpty(testIds)) {
    rawDataQuery.whereRaw(`uta.report_key IN (${testIds.map((t) => `'${t}'`)})`)
  }
  rawDataQuery.groupBy('uta.test_id').as('rq')
 
  return rawDataQuery
}
 
export const getSchoolAverage = (
  testIds: string[],
  districtId: string,
  assignedBy: string,
  userId: string,
  studentId = undefined,
  termId = undefined
) => {
  const redshift = getRedshiftInstance() || _redshift
  const rawDataQuery = redshift({
    uta: 't_user_test_activity',
  })
    .select('uta.school_id')
    .select('uta.test_id')
    .select(redshift.raw(`sum(score) / count(score) as school_avg`))
    .select(
      redshift.raw(
        `(sum(score) / nvl(nullif(sum(max_score),0),1))*100 as school_avg_perf`
      )
    )
    .where({
      'uta.progress_status': 1,
      'uta.district_id': districtId,
      'uta.archived': 0,
    })
  if (assignedBy === assignedByStatus.ME) {
    rawDataQuery.where('uta.assigned_by', userId)
  }
  if (!isEmpty(testIds)) {
    rawDataQuery.whereRaw(`uta.report_key IN (${testIds.map((t) => `'${t}'`)})`)
  }
  if (!isEmpty(studentId) && !isEmpty(termId)) {
    rawDataQuery.whereIn(
      'school_id',
      redshift({
        uta1: 't_user_test_activity',
      })
        .distinct('school_id')
        .where({ student_id: studentId, term_id: termId })
    )
  }
  rawDataQuery.groupBy('uta.test_id', 'uta.school_id').as('rq')
 
  return rawDataQuery
}

export const appendGroupQuery = (
  rawDataQuery,
  teacherIds,
  groupIds,
  classIds,
  table = 'uta',
  column = 'student_id'
) => {
  const redshift = getRedshiftInstance() || _redshift
  const studentQuery = redshift('t_enrollments')
    .distinct('user_id')
    .whereRaw(
      `group_id IN (${[...groupIds, ...classIds].map((g) => `'${g}'`)})`
    )
    .where('user_role', userRoles.STUDENT)
 
  rawDataQuery.whereIn(`${table}.${column}`, studentQuery)
}
 
export const appendGroupQueryWithGroupId = (
  rawDataQuery,
  groupIds,
  classIds,
  table = 'uta'
) => {
  const redshift = getRedshiftInstance() || _redshift
  const studentQuery = redshift({ er: 't_enrollments' })
    .distinct('group_id')

  rawDataQuery.whereIn(`${table}.group_id`, studentQuery)
}
 
export const validateAndGetSchoolIds = async (
  userId: string,
  role: string,
  schoolIds: string[]
) => {
  let processedSchoolIds: string[] = []
  if (role === userRoles.SCHOOL_ADMIN) {
    const userModel = new UserModel()
    const user = await userModel
      .getById(userId, {
        institutionIds: 1,
      })
      .read('secondaryPreferred')
      .lean()
    if (user) {
      const { institutionIds = [] } = user
      // converting ObjectId to string for easier comparison
      const stringSchoolIds = institutionIds.map((el) => `${el}`)
      // checking if school admin has access to incoming schools
      if (schoolIds.length && stringSchoolIds.length) {
        processedSchoolIds = schoolIds.filter((s) =>
          stringSchoolIds.includes(s)
        )
      }
      if (!processedSchoolIds.length && stringSchoolIds.length) {
        // otherwise use all his schools
        processedSchoolIds = stringSchoolIds
      }
    }
  } else if (schoolIds.length) {
    // if user is not school admin, then return as is
    processedSchoolIds = schoolIds
  }
  return processedSchoolIds
}
 
export const getQuestionLabelMap = (
  testItemIds,
  testItemsMap,
  testItemQuestionIds,
  testIsDocBased
) => {
  const qLabelMap = {}
  if (testIsDocBased) {
    // unwind questions from testItems in doc based test
    const questions = flatMap(testItemIds, (itemId) =>
      testItemsMap[itemId].data.questions.filter((question) =>
        testItemQuestionIds.includes(question.id)
      )
    )
    // unique numbering for each question
    questions.forEach((ques, index) => {
      Object.assign(
        qLabelMap,
        getQuestionLabels(index, { 'data.questions': [ques] })
      )
    })
  } else {
    testItemIds.forEach((itemId, index) => {
      const questions = testItemsMap[itemId].data.questions.filter((question) =>
        testItemQuestionIds.includes(question.id)
      )
      // grouped numbering for each question in testItem
      Object.assign(
        qLabelMap,
        getQuestionLabels(index, { 'data.questions': questions })
      )
    })
  }
  return qLabelMap
}
 
export const findInCsvQuery = (sourceField, matches) =>
  `(${matches.map((m) => `${sourceField} LIKE '%,${m},%'`).join(' OR ')})`
 
export const getClassFilterQuery = (
  districtId,
  termId,
  courseIds: string[] = [],
  grades,
  subjects,
  teacherIds,
  userRole,
  schoolIds,
  classIds,
  useDistinct = true
) => {
  const redshift = getRedshiftInstance() || _redshift
  const classFilterQuery = redshift({
    tod: 't_org_data',
  })
  if (useDistinct) classFilterQuery.distinct('tod.group_id')
  else classFilterQuery.select('tod.group_id')
 
  classFilterQuery.where({
    'tod.district_id': districtId,
    'tod.term_id': termId,
  })
 
  courseIds = compact(courseIds)
  if (!isEmpty(courseIds)) {
    classFilterQuery.whereRaw(
      `tod.course_id IN (${courseIds.map((s) => `'${s}'`)})`
    )
  }
 
  if (!isEmpty(grades)) {
    classFilterQuery.whereRaw(findInCsvQuery('tod.match_grades', grades))
  }
 
  if (!isEmpty(subjects)) {
    classFilterQuery.whereRaw(
      `tod.subject IN (${subjects.map((s) => `'${s}'`)})`
    )
  }
 
  if (!isEmpty(teacherIds)) {
    classFilterQuery.whereRaw(
      `tod.teacher_id IN (${teacherIds.map((t) => `'${t}'`)})`
    )
  } else {
    classFilterQuery.whereRaw('tod.teacher_id = tod.primary_teacher_id')
  }
 
  if (!isEmpty(schoolIds)) {
    classFilterQuery.whereRaw(
      `tod.school_id IN (${schoolIds.map((s) => `'${s}'`)})`
    )
  }
 
  if (!isEmpty(classIds)) {
    classFilterQuery.whereRaw(
      `tod.group_id IN (${classIds.map((s) => `'${s}'`)})`
    )
  }
 
  return classFilterQuery
}
