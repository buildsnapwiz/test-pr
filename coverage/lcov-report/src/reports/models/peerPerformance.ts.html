import _ from 'lodash'
import _redshift, { getRedshiftInstance } from '../../services/redshift'
import { status as testActivityStatus } from '../../const/testActivity'
import { assignedBy as assignedByStatus } from '../../const/assignment'
import { appendGroupQuery, findInCsvQuery } from './commons'
import { userRoles } from '../../const'
import { GROUP_TYPES } from '../../const/group'
 
const { ABSENT, SUBMITTED } = testActivityStatus
 
export const getPeerPerformanceMetrics = (
  termId: string,
  testId: string,
  districtId: string,
  schoolIds: string[],
  teacherIds: string[],
  subjects: string[],
  grades: string[],
  courseId: string,
  classIds: string[],
  groupIds: string[],
  assessmentTypes: string[],
  userRole: string,
  assignedBy: string,
  userId: string
) => {
  const redshift = getRedshiftInstance() || _redshift
  const assignmentQuery = redshift({
    ta: 't_assignments',
  })
    .distinct('ta.group_id')
    .where({
      'ta.district_id': districtId,
      'ta.term_id': termId,
      'ta.report_key': testId,
    })
  if (!_.isEmpty(assessmentTypes)) {
    assignmentQuery.whereIn('ta.test_type', assessmentTypes)
  }
  const classFilterQuery = redshift({
    tod: 't_org_data',
  })
    .distinct(
      'tod.group_id',
      'tod.group_name',
      'tod.gtype',
      'tod.school_id',
      'tod.school_name',
      'tod.teacher_id',
      'tod.teacher_name'
    )
    .where({
      'tod.district_id': districtId,
      'tod.term_id': termId,
    })
  if (!_.isEmpty(teacherIds)) {
    classFilterQuery.whereRaw(
      `tod.teacher_id IN (${teacherIds.map((t) => `'${t}'`)})`
    )
  } else {
    classFilterQuery.whereRaw('tod.teacher_id = tod.primary_teacher_id')
  }
  if (!_.isEmpty(courseId)) {
    classFilterQuery.where('tod.course_id', courseId)
  }
 
  if (!_.isEmpty(grades)) {
    classFilterQuery.whereRaw(findInCsvQuery('tod.match_grades', grades))
  }
 
  if (!_.isEmpty(subjects)) {
    classFilterQuery.whereRaw(
      `tod.subject IN (${subjects.map((s) => `'${s}'`)})`
    )
  }
 
  // remove groupIds to break uta and group direct link
  // 1. when only groupId is selected: classFilterQuery will get all classes+groups and appendGroupQuery will get the students for the selected groupId
  // 2. when only classIds is selected: classFilterQuery will get classes and appendGroupQuery is not applied
  // 3. when both are selected: classFilterQuery will get classes for the selected classId and appendGroupQuery will get the students for the selected groupId
  if (!_.isEmpty(classIds)) {
    classFilterQuery.whereRaw(
      `tod.group_id IN (${classIds.map((c) => `'${c}'`)})`
    )
  }
  classFilterQuery.whereRaw(`
    case tod.gtype 
      when '${GROUP_TYPES.CLASS}' then tod.group_id in (${assignmentQuery})
		  else true 
    end`)
  classFilterQuery.as('cf')
 
  const rawDataQuery = redshift()
    .select(
      'cf.*',
      'uta.student_id',
      'uta.score as total_score',
      'uta.max_score',
      'uta.progress_status'
    )
    .from(classFilterQuery)
    .joinRaw(`,"t_user_test_activity" as "uta"`)
    .whereRaw('uta.group_id = cf.group_id')
    .where({
      'uta.term_id': termId,
      'uta.report_key': testId,
      'uta.archived': 0,
    })
    .whereIn('uta.progress_status', [ABSENT, SUBMITTED])
 
  if (assignedBy === assignedByStatus.ME) {
    rawDataQuery.where('uta.assigned_by', userId)
    assignmentQuery.where('ta.assigned_by', userId)
  }
 
  if (!_.isEmpty(groupIds)) {
    // remove classIds so that it returns only students from groups
    appendGroupQuery(rawDataQuery, teacherIds, groupIds, [])
  } else if (!_.isEmpty(classIds) || !_.isEmpty(groupIds)) {
    rawDataQuery.whereRaw(
      `uta.group_id IN (${[...classIds, ...groupIds].map((g) => `'${g}'`)})`
    )
  }
  if (!_.isEmpty(schoolIds)) {
    rawDataQuery.whereRaw(
      `uta.school_id IN (${schoolIds.map((s) => `'${s}'`)})`
    )
  }
  if (!_.isEmpty(districtId)) {
    rawDataQuery.where('uta.district_id', districtId)
  }
  rawDataQuery.as('rq')
 
  return redshift()
    .select(
      'rq.*',
      'usr.frl_status',
      'usr.gender',
      'usr.ell_status',
      'usr.race',
      'usr.iep_status',
      'usr.hispanic_ethnicity',
      'usr.ext_attributes'
    )
    .from(rawDataQuery)
    .joinRaw(`,"t_users" as "usr"`)
    .whereRaw('rq.student_id = usr.user_id')
}
