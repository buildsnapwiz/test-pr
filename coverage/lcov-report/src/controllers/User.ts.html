export const getDefaultGradesAndSubjects = async (user) => {
  const orgIds = [user._id]
  let orgTypes = [ORG_TYPE.TEACHER]
  const userRole = user.role
  if (userRole === userRoles.SCHOOL_ADMIN) {
    orgTypes = [ORG_TYPE.SCHOOL_ADMIN]
  } else if (userRole === userRoles.DISTRICT_ADMIN) {
    orgTypes = [ORG_TYPE.DISTRICT_ADMIN]
  }
  const interestedStandardList = (await interestedStandards.getUsingOrgIdsAndTypes(
    orgIds,
    orgTypes,
    {
      defaultGrades: 1,
      defaultSubjects: 1,
      autoShareGCAssignment: 1,
    }
  )) as InterestedStandardsSettingsTyping[]
  const defaultGrades: string[] = []
  const defaultSubjects: string[] = []
  let autoShareGCAssignment: boolean = false
  interestedStandardList.forEach(
    ({
      defaultGrades: dg = [],
      defaultSubjects: ds = [],
      autoShareGCAssignment: asGCA = false,
    }) => {
      defaultGrades.push(...dg)
      defaultSubjects.push(...ds)
      autoShareGCAssignment = asGCA
    }
  )
  return [
    [...new Set(defaultGrades)],
    [...new Set(defaultSubjects)],
    autoShareGCAssignment,
  ]
}
 
const processInterestedStandards = (
  interestedStandardList: any,
  orgTypes: string[]
) => {
  const orgTypeWiseCurriculums = _.groupBy(
    interestedStandardList,
    (intrstCurm) => {
      return intrstCurm.orgType
    }
  )
  const interestedCurriculumsObj = {}
  for (const p of orgTypes) {
    const orgCurriculums = orgTypeWiseCurriculums[p]
    if (!_.isEmpty(orgCurriculums)) {
      orgCurriculums.forEach((curriculumObj) => {
        if (!_.isEmpty(curriculumObj.curriculums)) {
          const subjectWiseCurriculum = _.groupBy(
            curriculumObj.curriculums,
            (cur) => {
              cur.orgType = p
              return cur.subject
            }
          )
          _.forOwn(subjectWiseCurriculum, (v, k) => {
            if (_.isEmpty(interestedCurriculumsObj[k])) {
              interestedCurriculumsObj[k] = subjectWiseCurriculum[k]
            } else {
              interestedCurriculumsObj[k].push(...subjectWiseCurriculum[k])
            }
          })
        }
      })
      if (!_.isEmpty(interestedCurriculumsObj)) break
    }
  }
  try {
    return {
      interestedCurriculums: _.flatMap(
        Object.values(interestedCurriculumsObj),
        (p) => p
      ),
    }
  } catch (e) {
    Logger.error(
      `Failed to process Interested Standards:: ${JSON.stringify(
        interestedCurriculumsObj
      )} :: `,
      e
    )
    return {}
  }
}
 
export const getInterestedCurriculums = async (
  orgData,
  userId,
  idsOnly = false,
  curatedIntrestedCurriculums = false
) => {
  const orgDistrictId = getCurrentDistrictId(
    _.pick(orgData, ['districtIds', 'currentDistrictId'])
  )
  const institutionIds = _.compact(orgData.institutionIds || []).map((el) => el)
  let orgIds = [orgDistrictId, ...institutionIds, userId]
  orgIds = _.compact(orgIds)
  // only userId match with one of the orgtype in teacher,school-admin,district-admin
  const orgTypes = [
    'teacher',
    'school-admin',
    'institution',
    'district-admin',
    'district',
  ]
 
  // @ts-ignore
  const { schoolAdminSettingsAccess = false } =
    (await districtPolicySettings.getUsingOrgIdAndType(
      orgDistrictId,
      ORG_TYPE.DISTRICT,
      {
        schoolAdminSettingsAccess: 1,
      }
    )) || {}
  if (!schoolAdminSettingsAccess) {
    _.pullAllBy(orgIds, institutionIds, (x) => `${x}`)
    _.pull(orgTypes, 'institution')
  }
  const interestedStandardList = await interestedStandards.getUsingOrgIdsAndTypes(
    orgIds,
    orgTypes,
    {
      [idsOnly ? 'curriculums._id' : 'curriculums']: 1,
      orgType: 1,
      orgId: 1,
      showAllStandards: 1,
    }
  )
 
  const interestedStandardsKeyByOrgId = _.keyBy(interestedStandardList, 'orgId')
  let showAllStandards = interestedStandardsKeyByOrgId[`${orgDistrictId}`]
    ? // @ts-ignore
      interestedStandardsKeyByOrgId[`${orgDistrictId}`].showAllStandards
    : false
  if (schoolAdminSettingsAccess) {
    const interestedStandardListForInstitutions = interestedStandardList.filter(
      (el) =>
        // @ts-ignore
        el.orgType === ORG_TYPE.INSTITUTION &&
        institutionIds.includes(`${el.orgId}`)
    )
    showAllStandards = interestedStandardListForInstitutions.some(
      // @ts-ignore
      (el) => el.showAllStandards === true
    )
    if (
      !showAllStandards &&
      interestedStandardListForInstitutions.length !== institutionIds.length
    ) {
      showAllStandards = interestedStandardsKeyByOrgId[`${orgDistrictId}`]
        ? // @ts-ignore
          interestedStandardsKeyByOrgId[`${orgDistrictId}`].showAllStandards
        : false
    }
  }
  if (idsOnly) {
    return interestedStandardList
  }
  if (curatedIntrestedCurriculums) {
    return processCuratedIntrestedCurriculums(
      userId,
      institutionIds,
      orgDistrictId,
      interestedStandardList
    )
  }
  return {
    showAllStandards,
    ...(await processInterestedStandards(interestedStandardList, orgTypes)),
  }
}
 
const processCuratedIntrestedCurriculums = (
  userId,
  schoolIds,
  districtId,
  interestedCurriculums
) => {
  // get interested curriculums by different orgType
  const interestedCurriculumIdsByUser = _.chain(interestedCurriculums)
    .filter((o) => `${o.orgId}` === `${userId}`)
    .flatMap((o) => (o && o.curriculums) || [])
    .map((o) => o && o._id)
    .filter((o) => o)
    .value()
  if (!_.isEmpty(interestedCurriculumIdsByUser)) {
    return interestedCurriculumIdsByUser
  }
  const subjectWiseinterestedCurriculumsObj = {}
  // Find school level intrested curriculumns
  const interestedCurriculumBySchool = _.chain(interestedCurriculums)
    .filter((o) => schoolIds.find((o1) => `${o1}` === `${o.orgId}`))
    .flatMap((o) => (o && o.curriculums) || [])
    .value()
  if (!_.isEmpty(interestedCurriculumBySchool)) {
    const subjectWiseCurriculum = _.groupBy(
      interestedCurriculumBySchool,
      (cur) => {
        return cur.subject
      }
    )
    Object.assign(subjectWiseinterestedCurriculumsObj, {
      ...subjectWiseCurriculum,
    })
  }
  // Find district level intrested curriculumns
  const interestedCurriculumByDistrict = _.chain(interestedCurriculums)
    .filter((o) => `${o.orgId}` === `${districtId}`)
    .flatMap((o) => (o && o.curriculums) || [])
    .value()
  if (!_.isEmpty(interestedCurriculumByDistrict)) {
    const subjectWiseCurriculum = _.groupBy(
      interestedCurriculumByDistrict,
      (cur) => {
        return cur.subject
      }
    )
    _.forOwn(subjectWiseCurriculum, (v, k) => {
      if (_.isEmpty(subjectWiseinterestedCurriculumsObj[k])) {
        subjectWiseinterestedCurriculumsObj[k] = v
      }
    })
  }
  const curatedIntrestedCurriculumIds: any = []
  _.forOwn(subjectWiseinterestedCurriculumsObj, (v, k) => {
    const subjectCurriculumsIds = _.chain(v)
      .map((o) => o._id)
      .filter((o) => o)
      .value()
    curatedIntrestedCurriculumIds.push(...subjectCurriculumsIds)
  })
  return curatedIntrestedCurriculumIds
}

      if (stdFirstLastName.length > 1) {
        lastName = stdFirstLastName[stdFirstLastName.length - 1]
        firstName = _.join(
          stdFirstLastName.slice(0, stdFirstLastName.length - 1),
          ' '
        )
      } else {
        firstName = student.trim()
      }
      // earlier districtId has been sent as a param and it was not
      // being consumed in the query, so removed sending it
      username = await getUniqueUserName(firstName, lastName, studentInfo.role)
    } else if (
      addMultipleStudentsType.LAST_NAME_FIRST_NAME_USER_FORMAT === provider
    ) {
      const stdLastFirstName = student.split(' ')
      if (stdLastFirstName.length > 1) {
        lastName = stdLastFirstName[0].trim()
        firstName = _.join(
          stdLastFirstName.slice(1, stdLastFirstName.length),
          ' '
        )
      } else {
        lastName = student.trim()
      }
      // earlier districtId has been sent as a param and it was not
      // being consumed in the query, so removed sending it
      username = await getUniqueUserName(firstName, lastName, studentInfo.role)
    } else if (
      addMultipleStudentsType.GOOGLE_USER_FORMAT === provider ||
      addMultipleStudentsType.MSO_USER_FORMAT === provider
    ) {
      // if group exist use school policy also to domain setting
      const emailDomain = student.substring(student.lastIndexOf('@') + 1)
      const districtPolicy = await districtPolicySettings.getUsingOrgIdAndType(
        districtId,
        ORG_TYPE.DISTRICT
      )
      const schoolPolicy = groupInfo
        ? await districtPolicySettings.getUsingOrgIdAndType(
            groupInfo.institutionId,
            ORG_TYPE.INSTITUTION
          )
        : null
      allowedValidDomain = await isDomainAllowedInOrgPolicy(
        districtPolicy,
        schoolPolicy,
        emailDomain,
        userRoles.STUDENT
      )
 
      if (!config.edulasticConstPassword.GOOGLE_OPEN_ID_PASSWORD) {
        throw 'GOOGLE_OPEN_ID_PASSWORD is Undefined in environment'
      }
      // TODO - this is not needed in V2
      // It was set in V1 to comply with some of the methods that were
      // used for authentication
      // Also await is not needed on the method hashPassword,
      // clean it up in all the references
      password = await hashPassword(
        config.edulasticConstPassword.GOOGLE_OPEN_ID_PASSWORD
      )
      studentInfo.email = student
      username = student
    }
    Object.assign(studentInfo, {
      username,
      email: username,
      firstName,
      lastName,
      password,
    })
    Object.assign(studentValidatedInfo, { username, firstName, lastName })
    let existingUser
    if (isValidEmailPattern(username)) {
      existingUser = await User.getByUsernameAndRole(username, studentInfo.role)
      /**
       * There is a chance that the user can have different values in the email & username attribute. ref: EV-30635
       * Adding fallback condition to find the user by email and role.
       * So there will not be any duplicate accounts can create with the same email.
       */
      if (!existingUser) {
        existingUser = await User.getByEmailAndRole(username, studentInfo.role)
      }
    } else {
      // we should be find the existing user within the district
      // the previous method getUserNameAndRole didn't
      // use the districtId in the query although passed in the params
      // There shouldn't be multiple users with the
      // same username and role within the district
      // so finding only one user
      // username never be empty. even if its empty it will create duplicate
      existingUser = await User.getByDistrictUserNameAndRole(
        districtId,
        username,
        studentInfo.role
      )
    }
    if (!existingUser) {
      if (allowedValidDomain) {
        let newUser
        const institutionIds: any = []
        if (!_.isEmpty(groupInfo)) {
          institutionIds.push(groupInfo.institutionId)
        } else if (institutionId) {
          institutionIds.push(new ObjectID(institutionId))
        }
        Object.assign(studentInfo, {
          institutionIds,
        })
        Object.assign(studentInfo, { status: Status.ACTIVE })
        if (isValidEmailPattern(username)) {
          // passing role along with username to query, because there can be
          // multiple users with same username and different roles
          newUser = await User.createOrUpdateByUserName(
            username,
            studentInfo,
            studentInfo.role
          )
        } else {
          // passing role along with username to query, because there can be
          // multiple users with same username and different roles within the district
          newUser = await User.createOrUpdateByDistrictAndUserName(
            districtId,
            username,
            studentInfo,
            studentInfo.role
          )
        }
        if (newUser) {
          const { _id: userId } = newUser
          studentValidatedInfo.userId = userId
          studentValidatedInfo.status = 'SUCCESS'
          newStudents.push(userId)
          enrollStudentsDetails.push(userId)
          // if current user adding him self as student. link both accounts
          if (
            isValidEmailPattern(username) &&
            !_.isEmpty(currentUser) &&
            currentUser.email === username &&
            currentUser.role === userRoles.TEACHER
          ) {
            await autoLinkUsersByIds(
              [userId, currentUser._id],
              currentUser.personId
            )
          }
        } else {
          studentValidatedInfo.status = 'ERROR_IN_SAVE'
        }
      } else {
        studentValidatedInfo.status = 'FAILED_DOMAIN_RESTRICTED'
        failedStudentsDetails.push(studentValidatedInfo)
      }
    } else {
      const {
        _id: userId,
        institutionIds = [],
        districtIds = [],
      } = existingUser
      // If add user to group then check the institutionIds and
      // update if current group institution is not exists with user.
      if (!_.isEmpty(groupInfo)) {
        const userInstitutionIds: any = institutionIds || []
        const userDistrictIds: any = districtIds || []
        let updateUser = false
        if (
          !userInstitutionIds
            .map((inst) => `${inst}`)
            .includes(`${groupInfo.institutionId}`)
        ) {
          existingUser.institutionIds = _.compact(
            _.uniqBy(
              [...userInstitutionIds, groupInfo.institutionId],
              (o) => `${o}`
            )
          )
          updateUser = true
        }
        // update if current group district is not exists with user.
        if (
          !userDistrictIds
            .map((dId) => `${dId}`)
            .includes(`${groupInfo.districtId}`)
        ) {
          existingUser.districtIds = _.compact(
            _.uniqBy([...userDistrictIds, groupInfo.districtId], (o) => `${o}`)
          )
          updateUser = true
        }
        if (updateUser) {
          await User.update(userId, existingUser)
        }
      }
      studentValidatedInfo.userId = userId
      studentValidatedInfo.status = 'FAILED_USER_EXISTS'
      enrollStudentsDetails.push(userId)
    }
    addedStudentDetails.push(studentValidatedInfo)
  }
  // call api to add students to group with new students
  // and students which require to enrol to a group.
  if (
    !_.isEmpty(groupInfo) &&
    (!_.isEmpty(enrollStudentsDetails) || !_.isEmpty(newStudents))
  ) {
    addedStudentDetails = await addMultipleStudentsToGroup(
      districtId,
      groupId,
      enrollStudentsDetails,
      newStudents,
      groupInfo
    )
 
    if (failedStudentsDetails && failedStudentsDetails.length > 0) {
      addedStudentDetails.push(...failedStudentsDetails)
    }
  }
  return addedStudentDetails
}
 
export const checkPremiumGradeSubject = async (premiumGradeSubject, userId) => {
  let premium = false
  if (premiumGradeSubject) {
    premium = !!premiumGradeSubject.find(
      (item) =>
        item.grade.toLowerCase() === 'all' &&
        item.subject.toLowerCase() === 'all'
    )
    if (!premium) {
      const groupList = await new GroupModel()
        .getAllByOwnerId(userId, true)
        .lean()
 
      if (groupList.length) {
        premium = groupList.some((group) => {
          const { grades, subject } = group
          return !!premiumGradeSubject.find(
            (item) =>
              (grades.includes(item.grade) ||
                (item.grade && item.grade.toLowerCase() === 'all')) &&
              (subject === item.subject ||
                (item.subject && item.subject.toLowerCase() === 'all'))
          )
        })
      }
    }
  }
  return premium
}

export const getUserSubscription = async (
  districtId,
  userId,
  schoolIds,
  districtIds,
  userRole
) => {
  const subscriptionModel = new OrgSubscription()
  let subscriptionData: any
  let districtSubscriptions: any
 
  if (districtId) {
    subscriptionData = await subscriptionModel.getUserSubscription(
      districtId,
      userId,
      schoolIds,
      districtIds
    )
    districtSubscriptions = await subscriptionModel.getDistrictSubscription([
      new ObjectID(districtId),
    ])
  }
 
  const calculatorProvider =
    (Array.isArray(districtSubscriptions) &&
      districtSubscriptions[0] &&
      districtSubscriptions[0].calculatorProvider) ||
    'EDULASTIC'
  const orgSubscriptions = {
    premium: false,
    enterprise: false,
    premiumGradeSubject: [],
    calculatorProvider,
  }
 
  if (!_.isEmpty(subscriptionData)) {
    const subscriptionRowsKeyed = _.keyBy(subscriptionData, '_id')
    if (
      subscriptionRowsKeyed[subscriptionType.ENTERPRISE] &&
      subscriptionRowsKeyed[subscriptionType.ENTERPRISE]
        .enterpriseOrPartialPremiumStatus > 0
    ) {
      Object.assign(orgSubscriptions, {
        enterprise: true,
      })
    } else if (
      subscriptionRowsKeyed[subscriptionType.PREMIUM] &&
      subscriptionRowsKeyed[subscriptionType.PREMIUM].active > 0
    ) {
      Object.assign(orgSubscriptions, {
        premium: true,
      })
      // if user has partial perminum subscription, send to FE,
      // its used to refresh user's data when group is created
      if (
        subscriptionRowsKeyed[subscriptionType.PARTIAL_PREMIUM] &&
        subscriptionRowsKeyed[subscriptionType.PARTIAL_PREMIUM]
          .enterpriseOrPartialPremiumStatus > 0
      ) {
        const subscription =
          subscriptionRowsKeyed[subscriptionType.PARTIAL_PREMIUM]
        Object.assign(orgSubscriptions, {
          premiumGradeSubject: _.flatten(subscription.gradeSubject),
        })
      }
    } else if (
      subscriptionRowsKeyed[subscriptionType.TRIAL_PREMIUM] &&
      subscriptionRowsKeyed[subscriptionType.TRIAL_PREMIUM].active > 0
    ) {
      Object.assign(orgSubscriptions, {
        premium: true,
      })
    } else if (
      subscriptionRowsKeyed[subscriptionType.PARTIAL_PREMIUM] &&
      subscriptionRowsKeyed[subscriptionType.PARTIAL_PREMIUM]
        .enterpriseOrPartialPremiumStatus > 0
    ) {
      const subscription =
        subscriptionRowsKeyed[subscriptionType.PARTIAL_PREMIUM]
 
      let premium = false
      if (userRole === userRoles.DISTRICT_ADMIN) {
        premium = subscription.adminPremium > 0
      } else if (userRole === userRoles.SCHOOL_ADMIN) {
        const ids = schoolIds.map((s) => `${s}`)
        premium = (subscription.institutionIds || []).some((x) =>
          ids.includes(`${x}`)
        )
      } else {
        premium = await checkPremiumGradeSubject(
          _.flatten(subscription.gradeSubject),
          userId
        )
      }
      Object.assign(orgSubscriptions, {
        premiumGradeSubject: _.flatten(subscription.gradeSubject),
        premium,
      })
    }
  }
  return orgSubscriptions
}

export const deleteStudent = async (
  districtId,
  userIds,
  statusReason = 'Explicit delete requested'
) => {
  const groupModel = new GroupModel()
  const enrollmentModel = new EnrollmentModel()
  const groupStudentCount = await enrollmentModel.Enrollment.aggregate([
    {
      $match: {
        'user._id': { $in: userIds },
      },
    },
    {
      $group: {
        _id: { groupId: '$group._id' },
        studentCount: { $sum: 1 },
      },
    },
    {
      $project: {
        groupId: '$_id.groupId',
        stdCount: '$studentCount',
        _id: 0,
      },
    },
  ])
  const groupUpdateDocs = []
  if (groupStudentCount) {
    groupStudentCount.forEach(({ groupId, stdCount }) => {
      groupUpdateDocs.push({
        filter: {
          // @ts-ignore
          districtId: new ObjectID(districtId),
          // @ts-ignore
          _id: groupId,
        },
        update: {
          $inc: {
            // @ts-ignore
            studentCount: -stdCount,
          },
        },
      })
    })
  }
  if (groupUpdateDocs.length > 0) {
    await groupModel.bulkWrite('updateMany', groupUpdateDocs, {})
  }
  await enrollmentModel.updateMany(
    { 'user._id': { $in: userIds } },
    { status: Status.ARCHIVED }
  )
  await User.updateMany(
    { _id: { $in: userIds } },
    {
      $set: {
        status: Status.ARCHIVED,
        statusReason,
        statusModifiedAt: Date.now(),
      },
    }
  )
}
 
export const moveUsers = async (
  districtId,
  destinationClassCode,
  sourceClassCode,
  userDetails
) => {
  const groupModel = new GroupModel()
  const enrollmentModel = new EnrollmentModel()
  const moveUsersResponse: any = []
  let status
  const destinationClass = await groupModel.getByCodeAndDistrictId(
    destinationClassCode,
    districtId
  )
  if (!destinationClass) {
    status = `Destination class doesn't exists with the class code ${destinationClassCode}`
  }
  const sourceClass = await groupModel.getByCodeAndDistrictId(
    sourceClassCode,
    districtId
  )
  if (!sourceClass) {
    status = `Source class doesn't exists with the class code ${sourceClassCode}`
  }
 
  if (!destinationClass || !sourceClass) {
    return moveUsersResponse.push({
      status,
    })
  }
 
  const { _id: sourceGroupId, institutionId: sourceInstitutionId } = sourceClass
  const {
    owners = [],
    primaryTeacherId,
    _id: destinationGroupId,
    districtId: destinationDistrictId,
    institutionId: destinationInstitutionId,
    institutionName: destinationInstitutionName,
    name: destinationGroupName,
  } = destinationClass
  const userEnrollments = await enrollmentModel.getUserIdByClassId(
    destinationGroupId
  )
  const userIds = userEnrollments.map((o) => o.user._id)
  const statusMap = userEnrollments.reduce((n, o) => {
    n[`${o.user._id}`] = o.status
    return n
  }, {})
  const usersInfo = await userModelInst.getByIds(userDetails, {
    email: 1,
    username: 1,
    role: 1,
    firstName: 1,
    lastName: 1,
    middleName: 1,
  })
  const userDetailsMap = usersInfo.reduce((n, o) => {
    n[o._id] = o
    return n
  }, {})
  const newlyAddedStudentIds: String[] = []
  for (const userId of userDetails) {
    const user = userDetailsMap[userId]
    if (
      userIds &&
      userIds.length > 0 &&
      userIds.map((uId) => uId.toString()).includes(userId)
    ) {
      status = 'User exists in the same class'
      if (statusMap[`${userId}`] === Status.ARCHIVED) {
        newlyAddedStudentIds.push(userId)
        await enrollmentModel.reAddStudent(
          user._id,
          new ObjectID(destinationGroupId)
        )
        status = 'User updated with active user in a class'
      }
    } else {
      const enrollment = {
        districtId: destinationDistrictId,
        institutionId: destinationInstitutionId,
        user: {
          _id: user._id,
          username: user.username,
          firstName: user.firstName,
          lastName: user.lastName,
          middleName: user.middleName,
        },
        group: {
          _id: destinationGroupId,
          name: destinationGroupName,
          code: destinationClassCode,
        },
        type: destinationClass.type,
        status: Status.ACTIVE,
        role: user.role,
      }
      const actionsArray: any[] = []
      if (user.role === userRoles.STUDENT) {
        newlyAddedStudentIds.push(user._id)
        actionsArray.push(
          enrollmentModel.create(enrollment),
          groupModel.addStudent(destinationGroupId, districtId),
          groupModel.updateStudentCount(
            sourceGroupId,
            -1,
            new ObjectID(districtId)
          ),
          enrollmentModel.deactivateUserEnrollment([user._id], sourceGroupId),
          assignClassAssignmentToNewStudents(
            districtId,
            destinationGroupId,
            1,
            [user._id]
          )
        )
      } else if (user.role === userRoles.TEACHER) {
        actionsArray.push(enrollmentModel.create(enrollment))
        // add teacher to group
        if (!primaryTeacherId)
          Object.assign(destinationClass, { primaryTeacherId: user._id })
        // @ts-ignore
        owners.push({
          id: user._id,
          email: user.email,
          name: _.without(
            [user.firstName, user.middleName, user.lastName],
            undefined,
            null,
            ''
          ).join(' '),
        })
        Object.assign(destinationClass, { owners })
        actionsArray.push(
          groupModel.update(destinationGroupId, districtId, destinationClass)
        )
 
        // get Test shared at school level
        const Tests =
          (await new Test().getTestByCreatedBySharedTypeAndSharedWithId(
            user._id,
            sourceInstitutionId,
            'SCHOOL'
          )) || []
        // deactivate teacher from old class
        actionsArray.push(
          enrollmentModel.deactivateUserEnrollment([user._id], sourceGroupId)
        )
        // update share with school ids
        for (const test of Tests) {
          const sharedWith = test.sharedWith || []
          const newSharedWith: {
            _id: ObjectID
            name: String
          }[] = []
          let doUpdate = false
          for (const shared of sharedWith) {
            const currentInstitutionId = `${shared._id}`
            if (
              `${destinationInstitutionId}` !== currentInstitutionId &&
              `${sourceInstitutionId}` === currentInstitutionId
            ) {
              newSharedWith.push({
                _id: destinationInstitutionId,
                name: destinationInstitutionName,
              })
              doUpdate = true
            } else {
              newSharedWith.push(shared)
            }
          }
          if (doUpdate) {
            Object.assign(test, { sharedWith: newSharedWith })
            await new Test().update(test._id, test)
          }
        }
      }
      await Promise.all(actionsArray)
      status = 'Existing user added to the class'
    }
    moveUsersResponse.push({
      username: user.username || '',
      firstName: user.firstName,
      lastName: user.lastName,
      role: userRoles.STUDENT,
      status,
    })
  }
  if (newlyAddedStudentIds.length) {
    await addStudentNotification(destinationClass, newlyAddedStudentIds)
  }
  return moveUsersResponse
}

export const createOrUpdateGoogleUser = async (user) => {
  const users = await User.User.find(
    {
      googleId: user.googleId,
      role: user.role,
    },
    { createdAt: 1, created: 1, status: 1 }
  )
    .sort({ _id: -1 })
    .lean()
  // sometimes due to db patches status is saved as floating point
  // find active user, if exists continue with it
  let _user = users.find((u) => parseInt(`${u.status}`, 10) === 1)
  if (!_user) {
    // if active user not found, pick the recently created user which is inactive
    ;[_user] = users
      .filter((u) => !(u.createdAt || u.created))
      .sort((u1, u2) => {
        if ((u1.createdAt || u1.created) > (u2.createdAt || u2.created)) {
          return -1
        }
        if ((u1.createdAt || u1.created) < (u2.createdAt || u2.created)) {
          return 1
        }
        return 0
      })
    if (!_user) {
      // fall back to pick the first one which is sorted on _id by recent to oldest
      ;[_user] = users
    }
  }
  if (_user) {
    return User.User.findOneAndUpdate(
      {
        _id: _user._id,
      },
      { $set: user },
      { upsert: true, new: true }
    )
  }
  const matchFilter = {
    googleId: user.googleId,
  }
  // Match role as well otherwise it will override existing role.
  if (!_.isEmpty(user.role)) {
    Object.assign(matchFilter, { role: user.role })
  }
  return User.User.findOneAndUpdate(
    matchFilter,
    { $set: user },
    { upsert: true, new: true }
  )
}
