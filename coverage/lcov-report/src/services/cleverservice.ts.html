const callCleverAPI = async ({
  axiosData,
  retries = 5,
  additionalInfo = {},
  retryCounter = 1,
  lambdaRequestId = 'NA',
}) => {
  // on error 429 "Rate limit exceeded" wait for X-RateLimit-Reset + 1 sec and then retry
  // try for 5 attempts then throw error if request is still failing
  try {
    return await axios(axiosData)
  } catch (error) {
    // 500, 502, 503	Clever API failure	Wait and retry request.
    // 429 Rate limit exceeded	Try again after rate limit period expires.
    // 504 Gateway Timeout
    if (
      [429, 500, 502, 503, 504].includes(
        _.get(error, 'response.status', undefined)
      )
    ) {
      const timeNow = Date.now()
      const resetTime = _.get(
        error,
        'response.headers.x-ratelimit-reset',
        timeNow / 1000
      )
      // resetTime is in milliseconds
      const randomNumber = _.random(5000, 20000) // random time in milliseconds between 5s and 20s
      const retryTime = Number(resetTime) * 1000 - timeNow + randomNumber
      if (retries > 0 && retryTime > 0) {
        await new Promise((r) => setTimeout(r, retryTime))
        ApiTraceLogger(
          'info',
          'CLEVER_SYNC_LOG_RETRY_INFO',
          { RequestID: lambdaRequestId },
          {},
          {
            message: `Retrying clever API call. Retry after: ${retryTime} milliseconds, x-ratelimit-reset: ${resetTime}, url: ${axiosData.url} current retry: ${retryCounter}`,
          }
        )
        retryCounter++
        return callCleverAPI({
          axiosData,
          retries: retries - 1,
          additionalInfo,
          retryCounter,
          lambdaRequestId,
        })
      }
      if (retries === 0 && retryTime > 0) {
        ApiTraceLogger(
          'error',
          'CLEVER_SYNC_LOG_RETRY_INFO',
          { RequestID: lambdaRequestId },
          {},
          {
            message: `All 5 retry for clever API is done and still no response received, so error being thrown. Retry after: ${retryTime} milliseconds, x-ratelimit-reset: ${resetTime}, url: ${
              axiosData.url
            } current retry: ${retryCounter} details:: ${JSON.stringify(
              additionalInfo
            )}`,
          }
        )
        error.message = `All 5 retry for clever API is done and still no response received, so error being thrown. Retry after: ${retryTime} milliseconds, x-ratelimit-reset: ${resetTime}, url: ${
          axiosData.url
        } current retry: ${retryCounter} details:: ${JSON.stringify(
          additionalInfo
        )}, Error message: ${error.message}`
        throw error
      } else {
        error.message = `Exception while calling clever APIs. Url:: ${
          axiosData.url
        } Details:: ${JSON.stringify(additionalInfo)}, Error message: ${
          error.message
        }`
        throw error
      }
    } else {
      error.message = `Exception while calling clever APIs. Url:: ${
        axiosData.url
      } Details:: ${JSON.stringify(additionalInfo)}, Error message: ${
        error.message
      }`
      throw error
    }
  }
}
 
const getTokenForAuthentication = async (districtId) => {
  const { clientId, clientSecretId, authUrl } = config.clever
  const authString = `${clientId}:${clientSecretId}`
  const authEncBytes = base64.encode(authString)
  let url = `${authUrl}?owner_type=district` // token for all district
  if (districtId) {
    url = `${authUrl}?owner_type=district&district=${districtId}` // token for specific district
  }
  const request = {
    method: 'get',
    url,
    headers: { Authorization: `Basic ${authEncBytes}` },
  }
  const districtToken = await callCleverAPI({
    axiosData: request,
    retries: 5,
    additionalInfo: {
      districtCleverId: districtId,
      note: 'Fetching access token for authorization',
    },
  })
  return districtToken
}

      }

        break
    }
  }
  return mergeAttribute
}

const checkEventDataSize = (result) => {
  const dataSize = Buffer.byteLength(JSON.stringify(result)) / (1024 * 1024)
  // Data limit check to process the limited size of data
  // 15 MB will include around 25 k event
  if (dataSize > DATA_LIMIT) {
    ApiTraceLogger(
      'error',
      'CLEVER_EVENT_DATA_LIMIT_EXCEED',
      {},
      new Error('Data size exceeded.'),
      {
        dataSize,
      }
    )
    return true
  }
  return false
}
 
export const getCleverData = async (
  uri,
  authToken,
  queryParams = {},
  additionalInfo = {},
  lambdaRequestId = 'NA'
) => {
  let results: any[] = []
  if (authToken) {
    let getData = true
    do {
      try {
        const reqData = {
          headers: {
            'X-CORRELATION-ID': lambdaRequestId,
            Authorization: `Bearer ${authToken}`,
          },
          method: 'GET',
          url: `${CLEVER_DOMAIN_URL}${uri}`,
          params: queryParams,
        }
        const resultSet = await callCleverAPI({
          axiosData: reqData,
          retries: 5,
          additionalInfo,
          lambdaRequestId,
        })
        const data = resultSet ? resultSet.data : {}
        if (data) {
          let actualDataList = _.map(data.data, 'data')
          if (
            !_.get(data, 'data[0].data') &&
            !_.isEmpty(actualDataList) &&
            actualDataList[0] === undefined
          ) {
            actualDataList = [data.data]
          }
          results = _.concat(results, actualDataList)
          if (data.links) {
            const nextDataRef = data.links.filter((o) => o.rel === 'next')
            if (nextDataRef.length === 1) {
              queryParams = {}
              const { uri: nextUri } = nextDataRef[0]
              uri = nextUri.replace('/v2.1', '')
 
              // check data size, after 15,000 events
              if (results?.length > 15000 && uri.indexOf('event') > -1) {
                const isSizeExceeded = checkEventDataSize(results)
                if (isSizeExceeded) {
                  getData = false
                }
              }
            } else {
              getData = false
            }
          }
        }
      } catch (error) {
        // if resource not found return empty data
        if (_.get(error, 'response.status', undefined) === 404) {
          ApiTraceLogger('info', 'DATA_INFO', {}, error, {
            message: `Resource not found. clever API :: ${CLEVER_DOMAIN_URL}${uri}`,
          })
          return []
        }
        ApiTraceLogger('error', 'SERVER_ERROR', {}, error, {
          message: `Error on clever API call. clever API :: ${CLEVER_DOMAIN_URL}${uri}`,
        })
        // adding logger for temporary user will be removed after debugging
        // logger.error(error, error.message);
        error.message = `Error thrown on clever API call. clever API :: ${CLEVER_DOMAIN_URL}${uri} Details:: ${JSON.stringify(
          additionalInfo
        )}, Error message: ${error.message}`
        throw error
      }
    } while (getData)
    if (results.length && uri === '/schools') {
      const cleverSchools = await schoolModel.getByCleverIds(
        results.map((cleverSchool) => cleverSchool.id),
        { stopSync: 1, cleverId: 1 }
      )
      const cleverSchoolMap = _.keyBy(cleverSchools, 'cleverId')
      results.forEach((cleverSchool) => {
        const { _id, stopSync = false } = cleverSchoolMap[cleverSchool.id] || {}
        Object.assign(cleverSchool, { stopSync }, _id ? { _id } : {})
      })
    }
  }
  return results
}

      await enrollmentModel.updateMany(
        {
          'user._id': existingUser._id,
          type: 'custom',
          status,
        },
        {
          status: status === Status.ARCHIVED ? Status.ACTIVE : Status.ARCHIVED,
          statusReason: `${
            status === Status.ARCHIVED ? 'UnArchiving' : 'Archiving'
          } by clever event id: ${eventId}`,
          statusModifiedAt: Date.now(),
        }
      )
      // increment count for all custom groups
      await groupModel.updateStudentCountForGroups(
        groupIds,
        status === Status.ARCHIVED ? 1 : -1
      )
    }
  }
}

) => {
  let modifiedUserDetails
  switch (userRole) {
    case 'teacher': {
      modifiedUserDetails = modifyCleverTeacherData(cleverUser, districtId)
      break
    }
    case 'student': {
      modifiedUserDetails = modifyCleverStudentData(cleverUser, districtId)
      break
    }
    case 'district-admin': {
      modifiedUserDetails = modifyCleverDistrictAdminData(
        cleverUser,
        districtId
      )
      break
    }
    case 'school-admin': {
      modifiedUserDetails = modifyCleverSchoolAdminData(cleverUser, districtId)
      break
    }
    default: {
      ApiTraceLogger(
        'info',
        'CLEVER_SYNC_LOG',
        { RequestID: lambdaRequestId },
        {},
        {
          message: `Clever sync user role is not supported. role: ${userRole} cleverId: ${cleverUser.id}`,
        }
      )
      modifiedUserDetails = null
      break
    }
  }
  ApiTraceLogger(
    'info',
    'CLEVER_SYNC_LOG',
    { RequestID: lambdaRequestId },
    {},
    {
      message: `Clever sync user data processing. role: ${userRole} cleverId: ${modifiedUserDetails.cleverId}`,
    }
  )
  let isUserUnArchiving = false
  if (modifiedUserDetails && modifiedUserDetails.cleverId) {
    const {
      credentials: { district_username: cleverUserName = '' } = {},
      cleverId: cleverUserId,
      role,
    } = modifiedUserDetails
    const userExistsWithId = false
    let { email: cleverEmail } = modifiedUserDetails
    cleverEmail = cleverEmail || cleverUserName || ''
    let existingUser = await usersModel
      .getByCleverId({ cleverId: cleverUserId }, {})
      .lean()
    if (existingUser) {
      ApiTraceLogger(
        'info',
        'CLEVER_SYNC_LOG',
        { RequestID: lambdaRequestId },
        {},
        {
          message: `User found by cleverId. role: ${userRole} cleverId: ${cleverUserId}`,
        }
      )
      if (existingUser.status === Status.ARCHIVED) {
        existingUser.status = Status.ACTIVE
        isUserUnArchiving = true
        if (eventId) {
          existingUser.statusReason = `Re-activating the user by clever event id: ${eventId}`
          existingUser.statusModifiedAt = Date.now()
        }
      }
    }
    if (!existingUser && isValidEmailPattern(cleverEmail)) {
      // find user by email and role
      try {
        existingUser = await usersModel.getByEmailAndRole(
          cleverEmail,
          userRole,
          {}
        )
        if (existingUser) {
          ApiTraceLogger(
            'info',
            'CLEVER_SYNC_LOG',
            { RequestID: lambdaRequestId },
            {},
            {
              message: `Found user by email, cleverId: ${existingUser.cleverId}, role: ${existingUser.role}, userId: ${existingUser._id}, email: ${existingUser.email}, username: ${existingUser.username}`,
            }
          )
        }
      } catch (error) {
        logger.error(
          `Error in finding the user with email :: ${cleverEmail} :: cleverID ::${cleverUserId} --Exception ::`,
          error
        )
      }
    }
    if (
      !existingUser &&
      cleverEmail &&
      userRole === userRoles.STUDENT &&
      !isValidEmailPattern(cleverEmail)
    ) {
      // find user by email, role, firstname, and lastname
      let nameFilter = {}
      if (modifiedUserDetails.firstName || modifiedUserDetails.lastName) {
        if (modifiedUserDetails.firstName) {
          nameFilter = {
            firstName: {
              $regex: `^${modifiedUserDetails.firstName}$`,
              $options: 'i',
            },
          }
        }
        if (modifiedUserDetails.lastName) {
          nameFilter = {
            ...nameFilter,
            ...{
              lastName: {
                $regex: `^${modifiedUserDetails.lastName}$`,
                $options: 'i',
              },
            },
          }
        }
      }
      if (_.isEmpty(nameFilter)) {
        ApiTraceLogger(
          'error',
          'CLEVER_SYNC_ERROR_LOG',
          { RequestID: lambdaRequestId },
          {},
          {
            message: `Student does not have either first name or last name, cleverId: ${modifiedUserDetails.cleverId}`,
          }
        )
        return null
      }
      try {
        ;[existingUser] =
          (await usersModel.getByFields(
            {
              email: cleverEmail,
              role: userRole,
              ...nameFilter,
            },
            {}
          )) || []
        if (!_.isEmpty(existingUser)) {
          ApiTraceLogger(
            'info',
            'CLEVER_SYNC_LOG',
            { RequestID: lambdaRequestId },
            {},
            {
              message: `Found student by email, role, fname, lname, cleverId: ${existingUser.cleverId}, role: ${existingUser.role}, userId: ${existingUser._id}, email: ${existingUser.email}, username: ${existingUser.username}`,
            }
          )
        }
      } catch (error) {
        logger.error(
          `Error in finding the user with username :: ${cleverUserName} :: cleverID ::${cleverUserId} --Exception ::`,
          error
        )
      }
    }
    // check if user is allowed in existing user's district
    if (
      !_.isEmpty(existingUser) &&
      districtId &&
      userRole === userRoles.TEACHER
    ) {
      // check if existing user is allowed for district enrollment
      const isDistrictEnrollmentAllowed = await validateAndDisableExistingUser(
        districtId,
        existingUser,
        flowType,
        multiDistrictErrorStatus.UNRESOLVED,
        allowUserDisabling
      )
      if (!isDistrictEnrollmentAllowed) {
        if (!allowUserDisabling) {
          if (flowType === 'Clever SSO') {
            return {
              error: 'Forbidden',
              statusCode: 403,
              message: `Login restricted using this district url. Please use app.edulastic.com to login`,
            }
          }
        } else {
          existingUser = null
        }
      }
    }
    // TODO: use flag for is district enrollment allowed instead of setting user to null, and see impact
    if (!existingUser) {
      ApiTraceLogger(
        'info',
        'CLEVER_SYNC_LOG',
        { RequestID: lambdaRequestId },
        {},
        {
          message: `User not exists. creating new user. cleverId: ${modifiedUserDetails.cleverId}, role: ${modifiedUserDetails.role}`,
        }
      )

      if (!modifiedUserDetails.username) {
        let username = cleverUserName
        // if clever username and email not exist
        // add cleverId as username
        if (!cleverUserName && !cleverEmail) {
          username = modifiedUserDetails.cleverId
        } else if (!cleverUserName && cleverEmail) {
          // adding email as username if no clever username
          username = cleverEmail
        }
        Object.assign(existingUser, {
          username,
        })
      }
      if (!modifiedUserDetails.email) {
        let email = cleverEmail
        // if clever username and email not exist
        // add cleverId as email
        if (!cleverUserName && !cleverEmail) {
          email = modifiedUserDetails.cleverId
        } else if (
          cleverUserName &&
          !cleverEmail &&
          isValidEmailPattern(cleverUserName)
        ) {
          // adding username as email if no clever email exist and username is in email format

        }

          email,

      )
      // update fields of existing user
      const { institutionIds: existingUserInstitutionIds = [] } = existingUser
      if (eventId && existingUser.status !== Status.ACTIVE) {
        existingUser.statusReason = `Re-activating the user by clever event id: ${eventId}`
        existingUser.statusModifiedAt = Date.now()
      }
      Object.assign(
        existingUser,
        _.pickBy(
          modifiedUserDetails,
          (v, k) => !['email', 'username'].includes(k)
        ),
        {
          districtIds: _.uniqBy(
            _.compact([...(existingUser.districtIds || []), districtId]),
            (o) => `${o}`
          ),
          institutionIds: _.uniqBy(
            _.compact([
              ...existingUserInstitutionIds,
              ...(modifiedUserDetails.institutionIds || []),

            (o) => `${o}`
          ),

          // if user exists with different role update the existing user role same as clever role
          role: userRole,
        }
      )
      // update current signup status based on existing user
      if (
        existingUser?.institutionIds?.length &&
        existingUser?.currentSignupState ===
          signUpState.ACCESS_WITHOUT_SCHOOL &&
        existingUser.role === userRoles.TEACHER
      ) {
        Object.assign(existingUser, { currentSignupState: signUpState.DONE })
      }
      const {
        email: existingUserEmail,
        username: existingUserName,
      } = existingUser
      // update email if no existing email or clever email changed
      if (!existingUserEmail || existingUserEmail !== cleverEmail) {
        let email = cleverEmail
        // if clever username and email not exist
        // add cleverId as email
        if (!cleverUserName && !cleverEmail) {
          email = modifiedUserDetails.cleverId
        } else if (
          cleverUserName &&
          !cleverEmail &&
          isValidEmailPattern(cleverUserName)
        ) {
          // adding username as email if no clever email exist and username is in email format
          email = cleverUserName
        }
        Object.assign(existingUser, {
          email,
        })
      }
      // update username if no existing username or clever username changed
      if (!existingUserName || existingUserName !== cleverUserName) {
        let username = cleverUserName
        // if clever username and email not exist
        // add cleverId as email
        if (!cleverUserName && !cleverEmail) {
          username = modifiedUserDetails.cleverId
        } else if (!cleverUserName && cleverEmail) {
          // adding username as email if no clever username exists
          username = cleverEmail
        }
        Object.assign(existingUser, {
          username,
        })
      }
    }
    if (existingUser) {
      // populate districts for student
      if (role === userRoles.STUDENT) {
        Object.assign(existingUser, {
          districtIds: uniqBy(
            _.compact([...(existingUser.districtIds || []), districtId]),
            (o) => `${o}`
          ),
        })
      }
      if (existingUser._id) {
        let { personId } = existingUser
        if (_.isEmpty(personId) && existingUser.cleverId) {
          // if no person id exists
          const existingCleverUsers = await usersModel.User.find(
            {
              cleverId: existingUser.cleverId,
              status: Status.ACTIVE,
            },
            { personId: 1 }
          )
          // if more than one active user exists with same cleverId
          // update all such user with new personId
          if (
            !_.isEmpty(existingCleverUsers) &&
            existingCleverUsers.length > 1
          ) {
            // if multiple user exists with same cleverId, and any user is having personId
            // then save the same personId for current user also
            const { personId: pId = undefined } =
              existingCleverUsers.find((o) => !!o.personId) || {}
            personId = pId || uuidv4()
            Object.assign(existingUser, { personId })
            await usersModel.updateMany(
              { cleverId: existingUser.cleverId, status: Status.ACTIVE },
              { $set: { personId } }
            )
          }
        }
        ApiTraceLogger(
          'info',
          'CLEVER_SYNC_LOG',
          { RequestID: lambdaRequestId },
          {},
          {
            message: `Updating existing User. cleverId: ${existingUser.cleverId}, role: ${existingUser.role}, userId: ${existingUser._id}`,
          }
        )
        // update the existing user
        const updatedUsers = await usersModel.update(
          existingUser._id,
          existingUser
        )
        if (existingUser?.role === userRoles.STUDENT && isUserUnArchiving) {
          // update all custom group enrollment for the student
          await updateCustomGroupEnrollments(
            existingUser,
            Status.ARCHIVED,
            eventId
          )
        }
        return updatedUsers
      }
      ApiTraceLogger(
        'info',
        'CLEVER_SYNC_LOG',
        { RequestID: lambdaRequestId },
        {},
        {
          message: `Creating new user. cleverId: ${existingUser.cleverId}, role: ${existingUser.role}, userId: ${existingUser._id}`,
        }
      )
      // create the new user
      try {
        if (eventId) {
          existingUser.statusReason = `Creating user by clever event id: ${eventId}`
          existingUser.statusModifiedAt = Date.now()
        }
        return createOrGetUserByCleverId({
          userData: existingUser,
        })
      } catch (err) {
        ApiTraceLogger(
          'error',
          'SERVER_ERROR',
          { RequestID: lambdaRequestId },
          err,
          {
            message: `Error on save or update clever user :: cleverId: ${existingUser.cleverId}, districtId: ${districtId}`,
          }
        )
      }
    }
  }
  return Promise.resolve(null)
}

const getDistrictData = async (response, districtCleverId, district) => {
  let success = false
  let message = cleverLogMessage.CLEVER_SYNC_IN_PROGRESS
  const districtId = district.id
  if (response != null && response.data.length > 0) {
    const district = response.data[0]
    if (district.owner && district.owner.id === districtCleverId) {
      const token = district.access_token
      if (district && district.syncEnabled === false) {
        message = cleverLogMessage.CLEVER_SYNC_DISABLED
      } else if (district) {
        await getDistrict(token)
        await syncDistrict(token, districtId, districtCleverId)
        success = true
      }
    } else {
      message = cleverLogMessage.CLEVER_OWNER_ID_CLEVER_ID_MISMATCH
    }
  } else {
    message = cleverLogMessage.CLEVER_DISTRICT_NOT_FOUND
  }
  await saveCleverLog({}, districtId, message)
  return { success, message }
}

const cleverSSOInstantLogin = async (
  code,
  state,
  lambdaRequestId = 'NA',
  referrer = undefined,
  districtId
) => {
  const cleverState =
    state && !state.includes('assignment')
      ? base64.decode(state).isCleverClassSyncFLow
      : ''
  if (code && !cleverState) {
    const { clientId, redirectUri, clientSecretId, authUrl } = config.clever
    const reqBody = {
      code,
      grant_type: 'authorization_code',
      redirect_uri: redirectUri,
    }
    if (clientId && clientSecretId) {
      const authString = `${clientId}:${clientSecretId}`
      const authEncBytes = base64.encode(authString)
      const reqData = {
        method: 'POST',
        url: `${authUrl}`,
        headers: { Authorization: `Basic ${authEncBytes}` },
        data: reqBody,
      }
      const accessToken = _.get(
        await callCleverAPI({
          axiosData: reqData,
          retries: 5,
          additionalInfo: {
            note: 'Fetching access token using code on clever sso',
          },
          lambdaRequestId,
        }),
        'data.access_token',
        ''
      )
      if (accessToken) {
        const cleverUserDetails = _.get(
          await getUserUsingToken(accessToken),
          'data.data',
          ''
        )
        const { id: cleverUserId, type: cleverUserRole } = cleverUserDetails
        ApiTraceLogger(
          'info',
          'DATA_INFO',
          {},
          {},
          {
            message: `user cleverId :: ${cleverUserId} role: ${cleverUserRole}`,
          }
        )
        if (cleverUserId) {
          let existingUser
          const existingUsers = await usersModel
            .getByFields({
              cleverId: cleverUserId,
            })
            .lean()
          if (!_.isEmpty(existingUsers)) {
            if (cleverUserRole === 'district_admin') {
              existingUser = existingUsers.find(
                ({ role }) => role === userRoles.DISTRICT_ADMIN
              )
            } else if (cleverUserRole === 'school_admin') {
              existingUser = existingUsers.find(
                ({ role }) => role === userRoles.SCHOOL_ADMIN
              )
            } else if (cleverUserRole === roles.TEACHER) {
              existingUser = existingUsers.find(
                ({ role }) => role === userRoles.TEACHER
              )
            } else if (cleverUserRole === roles.STUDENT) {
              existingUser = existingUsers.find(
                ({ role }) => role === userRoles.STUDENT
              )
            }
            if (existingUsers.length > 0 && !existingUser) {
              // if existing user but role do not match
              // Show notification to the user
              return {
                error: 'Forbidden',
                statusCode: 403,
                message:
                  'Your Clever account role does not match with the existing Edulastic account. Please contact support.',
              }
            }
          }
          if (existingUser) {
            const signOnPolicyError = await validateSignOnPolicy(
              existingUser,
              'cleverSignOn',
              districtId,
              !districtId // don't enforce policy if coming from District page
            )
            if (signOnPolicyError) {
              return {
                statusCode: 403,
                error: 'Forbidden',
                message: signOnPolicyError,
              }
            }
            const {
              role: existingUserRole,
              _id: existingUserId,
              status: existingUserStatus,
            } = existingUser
            const isUserAdmin =
              existingUserRole === userRoles.DISTRICT_ADMIN ||
              existingUserRole === userRoles.SCHOOL_ADMIN
            const isUserTeacher = existingUserRole === userRoles.TEACHER
            if (!(isUserAdmin || isUserTeacher)) {
              if (
                existingUserRole === userRoles.STUDENT &&
                existingUserStatus === Status.DISABLE
              ) {
                return {
                  error: 'Forbidden',
                  statusCode: 403,
                  message:
                    'You are not authorized to use edulastic, Please contact your teacher.',
                }
              }
              const userEnrollments = await enrollmentModel.getByFields(
                {
                  'user._id': existingUserId,
                },
                { 'group._id': 1 }
              )
              const groupIds = userEnrollments.map((o) => o.group._id)
              if (
                !groupIds.length ||
                !(
                  await groupModel.getGroupByIdListAndStatus(
                    groupIds,
                    Status.ACTIVE,
                    { _id: 1 }
                  )
                ).length
              ) {
                return {
                  error: 'Forbidden',
                  statusCode: 403,
                  message: 'This user type is not authorized to use edulastic',
                }
              }
            } else if (isUserTeacher) {
              const updateInfo = {
                cleverAccessToken: accessToken,
              }
              // check if user is allowed to add multiple district
              if (!_.isEmpty(existingUser) && districtId) {
                // check if existing user is allowed for district enrollment
                const isDistrictEnrollmentAllowed = await validateAndDisableExistingUser(
                  districtId,
                  existingUser,
                  'Clever SSO',
                  multiDistrictErrorStatus.UNRESOLVED,
                  false
                )
                if (!isDistrictEnrollmentAllowed) {
                  // if its clever SSO return error with proper message here
                  return {
                    error: 'Forbidden',
                    statusCode: 403,
                    message: `Login restricted using this district url. Please use app.edulastic.com to login`,
                  }
                }
                Object.assign(updateInfo, {
                  districtIds: _.compact(
                    _.uniqBy(
                      [
                        ...(existingUser.districtIds || []),
                        new ObjectId(`${districtId}`),
                      ],
                      (o) => `${o}`
                    )
                  ),
                })
              }
              await usersModel.update(existingUserId, updateInfo)
            }
          } else if (cleverUserRole === userRoles.TEACHER) {
            /**
             *  Only INSTRUCTOR user type is supported for provisioning
             */
            Object.assign(cleverUserDetails, {
              currentSignUpState: signUpState.ACCESS_WITHOUT_SCHOOL,
              status: Status.ACTIVE,
              cleverAccessToken: accessToken,
              referrer,
            })
            const result = await saveORUpdateUserDetails(
              cleverUserDetails,
              undefined,
              districtId,
              userRoles.TEACHER,
              lambdaRequestId,
              undefined,
              'Clever SSO',
              false
            )
            if (result.error) {
              return result
            }
            existingUser = (await usersModel
              .getOneByFields({
                cleverId: cleverUserId,
                role: userRoles.TEACHER,
                status: Status.ACTIVE,
              })
              .lean()) as UserType
          } else {
            return {
              error: 'Forbidden',
              statusCode: 403,
              message:
                'User not yet authorized to use Edulastic. Please contact your district administrator!',
            }
          }
          return { ...existingUser }
        }
      } else {
        throw new Error('Invalid user')
      }
    } else {
      throw new Error('Clever not Configured ')
    }
  } else {
    throw new Error('Invalid request received')
  }
}

const cleverSyncSchoolsService = async (
  districtCleverId,
  schoolCleverIds,
  district,
  req
) => {
  const response = await getTokenForAuthentication(districtCleverId)
  const responseData = _.get(response, 'data.data', [])
  let success = false
  let message = cleverLogMessage.CLEVER_SYNC_IN_PROGRESS
  const districtId = district.id
  if (!_.isEmpty(responseData)) {
    const district = responseData[0]
    if (district.owner && district.owner.id === districtCleverId) {
      const token = district.access_token
      if (district && district.syncEnabled === false) {
        message = cleverLogMessage.CLEVER_SYNC_DISABLED
      } else if (district) {
        // get term and course for district and save them before processing each school
        // no need to save term, this will be used
        // only to get the end date of sections and term name if needed
        const { Terms: districtTerms } = await getTermUsingDistrictId(
          token,
          districtId,
          req.RequestID
        )
        // convert district terms to map
        const terms = districtTerms.reduce((acc, termData) => {
          acc[termData.id] = {
            name: termData.name,
            id: termData.id,
            end_date:
              termData.end_date && commons.getTimeStamp(termData.end_date),
            start_date:
              termData.start_date && commons.getTimeStamp(termData.start_date),
          }
          return acc
        }, {})
        // fetch the course at district level
        await getCourseUsingDistrictId(token, districtId, req.RequestID)
        await syncSchools(
          token,
          districtId,
          districtCleverId,
          schoolCleverIds,
          terms,
          'Clever School Sync'
        )
        success = true
      }
    } else {
      message = cleverLogMessage.CLEVER_OWNER_ID_CLEVER_ID_MISMATCH
    }
  } else {
    message = cleverLogMessage.CLEVER_DISTRICT_NOT_FOUND
  }
  await saveCleverLog(req, districtId, message)
  return { success, message }

const processMergeCSVCleverIds = async (
  districtId,
  mergeType,
  fileRows: Array<Array<String>>
) => {
  let result: any = {}
  const resultData: any = []
  let cleverRowIndex
  let eduRowIndex
  try {
    const usersType = ['stu', 'tch', 'sa', 'da']
    for (const [index, fileRow] of fileRows.entries()) {
      if (index === 0) {
        const headerColumns = fileRow
        for (const [colIndex, headerColumn] of headerColumns.entries()) {
          if (mergeType === 'sch') {
            // Header columns as per existing csv columns
            if (headerColumn === 'school_id') {
              eduRowIndex = colIndex
            } else if (headerColumn === 'id') {
              cleverRowIndex = colIndex
            }
          } else if (mergeType === 'cls') {
            if (headerColumn === 'edu_class_section_id') {
              eduRowIndex = colIndex
            } else if (headerColumn === 'clever_id') {
              cleverRowIndex = colIndex
            }
          } else if (usersType.includes(mergeType)) {
            if (headerColumn === 'user_id') {
              eduRowIndex = index
            } else if (headerColumn === 'id') {
              cleverRowIndex = index
            }
          }
        }
      } else if (cleverRowIndex >= 0 && eduRowIndex >= 0) {
        const cleverId = `${fileRow[cleverRowIndex].trim()}`
        const eduId = `${fileRow[eduRowIndex].trim()}`
        try {
          if (mergeType === 'sch') {
            await schoolModel.updatecleverId(eduId, cleverId)
          } else if (mergeType === 'cls') {
            await groupModel.updatecleverId(eduId, cleverId)
          } else if (usersType.includes(mergeType)) {
            await usersModel.updatecleverId(eduId, cleverId)
          }
          resultData.push({
            edulasticId: eduId,
            cleverId,
            status: 'success',
          })
        } catch (error) {
          await saveCleverLog(
            {},
            districtId,
            `Merge failed, merge type: ${mergeType},\
            edulastic_id: ${eduId}, clever_id: ${cleverId}`
          )
          resultData.push({
            edulasticId: eduId,
            cleverId,
            status: 'fail',
          })
        }
      }
    }
    await saveCleverLog({}, districtId, cleverLogMessage.CSV_UPDATE_SUCCESSFUL)
    result = {
      message: cleverLogMessage.CSV_UPDATE_SUCCESSFUL,
      status: 'success',
      data: resultData,
    }
  } catch (e) {
    await saveCleverLog(
      {},
      districtId,
      cleverLogMessage.FILE_PARSE_MERGE_FAILED
    )
    result.push({
      status: 'error',
      message: cleverLogMessage.FILE_PARSE_MERGE_FAILED,
    })
  }
  return result
}

      }

          (await schoolModel.getByIds([entityId], { districtId: 1 })) || []

          )

          )

const createEnrollmentForGroupUsers = async (
  cleverSection,
  districtId,
  multipleDistrictUser = []
) => {
  const { students = [], teachers = [], id } = cleverSection
  // get clever group by clever id
  const group = await groupModel.getAllGroupByDistrictIdAndCleverId(
    id,
    districtId
  )
  let userIds = []
  if (!_.isEmpty(multipleDistrictUser)) {
    userIds = [...students, ...teachers].filter(
      (id) => !multipleDistrictUser.includes(id)
    )
  } else {
    userIds = [...students, ...teachers]
  }
  let users: UserType[] = []
  if (group && !_.isEmpty(userIds)) {
    // get all user by cleverId (teachers and students)
    users = await usersModel.getByFields({ cleverId: { $in: userIds } }).lean()
    const createEnrollmentList: Promise<enrollment | null | void>[] = []
    const existingEnrollments = await getGroupEnrollments(
      group._id,
      districtId,
      { 'user._id': 1, status: 1 }
    )
    const enrolledUserIds = existingEnrollments
      .filter((o) => o.status === Status.ACTIVE)
      .map((o) => `${o.user._id}`)
    let newlyAddedStudentCount = 0
    const newlyAddedStudentIds: String[] = []
    // we are not activating existing archive enrollments.
    // When we do handle createOrUpdateUtaForReEnrollements method with reActivated student Ids.
    users.forEach((user) => {
      if (_.indexOf(enrolledUserIds, `${user._id}`) === -1) {
        createEnrollmentList.push(createUserEnrollment(districtId, group, user))
        if (user.role === 'student') {
          newlyAddedStudentIds.push(user._id)
          newlyAddedStudentCount++
        }
      }
    })
    if (newlyAddedStudentIds.length > 0) {
      createEnrollmentList.push(
        addStudentNotification(group, newlyAddedStudentIds)
      )
    }
    await Promise.all(createEnrollmentList)
    // reset student count in group
    await updateStudentCountInGroup(group._id, districtId, group.studentCount)
 
    // assign assignment to newly enrolled user
    if (newlyAddedStudentIds.length) {
      await assignClassAssignmentToNewStudents(
        districtId,
        group._id,
        newlyAddedStudentCount,
        newlyAddedStudentIds
      )
    }
  }
  // archive the existing enrollment for the user
  // if user not exists in clever class anymore
  // update the student count in class
 
  if (group) {
    // get current active enrollments
    const activeStudentEnrollments = await enrollmentModel.getByFields({
      'group._id': group._id,
      status: Status.ACTIVE,
      role: userRoles.STUDENT,
    })
    // get current active student ids
    const activeStudentIds = activeStudentEnrollments.map((o) => o.user._id)
 
    // get clever user from active enrolled student
    const activeCleverStudents = await usersModel.getByFields(
      {
        _id: { $in: activeStudentIds },
        role: userRoles.STUDENT,
        status: Status.ACTIVE,
        cleverId: { $exists: true },
      },
      {
        cleverId: 1,
      }
    )
    const activeStudentCleverId = activeCleverStudents
      .filter((o) => !!o.cleverId)
      .map((o) => o.cleverId)
    // find the clever user, no longer enrolled in current group
    let studentCleverIds
    if (users) {
      studentCleverIds = _.differenceWith(
        activeCleverStudents.map((o) => o.cleverId),
        users.map((o) => o.cleverId),
        (cId1, cId2) => `${cId1}` === `${cId2}`
      )
    } else {
      studentCleverIds = activeStudentCleverId
    }
    if (!_.isEmpty(studentCleverIds)) {
      const removeEnrolledUserIds = activeCleverStudents
        .filter((o) => studentCleverIds.includes(o.cleverId))
        .map((o) => o._id)
      await enrollmentModel.updateMany(
        {
          'user._id': { $in: removeEnrolledUserIds },
          'group._id': group._id,
        },
        {
          status: Status.ARCHIVED,
        }
      )
 
      await updateUtaForRemovedStudents(
        districtId,
        removeEnrolledUserIds.map((sId) => `${sId}`),
        `${group._id}`,
        utaArchiveReason.UN_ENROLLED
      )
 
      // update group student count
      const count = activeCleverStudents.length - removeEnrolledUserIds.length
      await groupModel.updateStudentCountWithNewCount(
        group._id,
        count,
        districtId
      )
    }
  }
}

const processCleverEvents = async ({ lambdaRequestId, ...data }) => {
  const {
    cleverEvent: eventData,
    edulasticDistrictId,
    rosterSyncConfig,
    schools,
    accessToken,
    terms,
  } = data
  const {
    type,
    data: { object, previous_attributes: prevAttributes },
  } = eventData || {}
  const { id: dataObjectId, district: districtCleverId } = object || {}
  const { school: previousSchoolCleverId } = prevAttributes || {}
  if (type) {
    // create or update event
    const eventTypeParts = type.split('.')
    const resource = eventTypeParts.length > 0 ? eventTypeParts[0] : ''
    const modification = eventTypeParts.length > 1 ? eventTypeParts[1] : ''
    if (object) {
      const eventId = eventData.id
      const eventInfo = `EventId: ${eventData.id}, action: ${resource}, entityCleverId: ${object.id} modification: ${modification}, districtId: ${edulasticDistrictId}, dCleverId: ${districtCleverId}`
      try {
        const resourceData = { ...object }
        ApiTraceLogger(
          'info',
          'CLEVER_DELTA_SYNC_LOG',
          { RequestID: lambdaRequestId },
          {},
          {
            message: `Running event. ${eventInfo}`,
          }
        )
        switch (resource) {
          case 'schools': {
            // save school data
            await saveSchoolData(resourceData, edulasticDistrictId, schools)
            await saveLastSyncCleverId(
              eventData,
              edulasticDistrictId,
              districtCleverId
            )
            break
          }

                    )
                  }
                  await updateUtaForRemovedStudents(
                    edulasticDistrictId,

                    `${group._id}`,

                  )
                }

                  lambdaRequestId,

                    group.id,
                    newStudentCounter,
                    newlyAddedStudentIds
                  )
                }

                eventData,
                edulasticDistrictId,
                districtCleverId
              )
            }
            break
          }
          default:

        }
        ApiTraceLogger(
          'info',
          'CLEVER_DELTA_SYNC_LOG',
          { RequestID: lambdaRequestId },
          {},
          {
            message: `Event completed successfully. ${eventInfo}`,
          }
        )
      } catch (e) {
        ApiTraceLogger(
          'error',
          'CLEVER_DELTA_SYNC_ERROR',
          { RequestID: lambdaRequestId },
          e,
          {
            message: `Error while running event. ${eventInfo}`,
          }
        )
      }
    }
  }
}

const moveClassSection = async (
  resourceData,
  edulasticDistrictId,
  previousSchoolCleverId
) => {
  const { id: sectionCleverId, school: schoolCleverId, sis_id } = resourceData
  if (!_.isEmpty(resourceData)) {
    // get group by district and clever id
    const group = await groupModel.getAllGroupByDistrictIdAndCleverId(
      sectionCleverId,
      edulasticDistrictId
    )
    if (group) {
      const sourceSchool = await schoolModel.getByDistrictIdAndCleverId(
        previousSchoolCleverId,
        edulasticDistrictId
      )
      const destinationSchool = await schoolModel.getByDistrictIdAndCleverId(
        schoolCleverId,
        edulasticDistrictId
      )
      if (
        sourceSchool &&
        destinationSchool &&
        `${sourceSchool._id}` !== `${destinationSchool._id}`
      ) {
        const { _id: destinationSchoolId, name } = destinationSchool
        // update class section's school id and sis_id
        Object.assign(group, {
          institutionId: destinationSchoolId,
          institutionName: name,
          sisId: sis_id,
        })
        await groupModel.update(group._id, group.districtId, group)
        const enrollments = await enrollmentModel.getUserIdByClassId(
          group._id,
          { 'user._id': 1 }
        )
        const userIds = enrollments.map((o) => o.user._id)
        // update user enrollments with destination institution id
        await enrollmentModel.updateMany(
          { 'group._id': group._id },
          { institutionId: destinationSchoolId }
        )
        // update user instituion with destination institution id
        await usersModel.updateMany(
          {
            _id: { $in: userIds },
            institutionIds: { $ne: destinationSchoolId },
          },
          { $push: { institutionIds: destinationSchoolId } }
        )
      }
    }
  }
}
