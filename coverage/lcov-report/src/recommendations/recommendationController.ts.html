const getUniqueRecommendations = (recommendations) => {
  const map: object = {}
  recommendations.forEach((el) => {
    map[commons.generateHash(`${el.standardIdentifier}|${el.description}`)] = el
  })
  return Object.values(map)
}
 
export const getDifferentiationStandards = async (
  standards: Array<number>,
  skillIdentifiers: Array<string>
) => {
  const recommendations: {
    [key: string]: Array<object>
  } = {
    review: [],
    challenge: [],
    practice: [],
  }
  if (standards.length) {
    const standardDetails = await standardModel._getByIdsFromCache(standards)
    if (standardDetails.length) {
      const supportedStandards = standardDetails.filter(
        (std) => std.curriculumId === SUPPORTED_CURRICULUM
      )
      if (supportedStandards.length) {
        const practiceStandards = supportedStandards.map(
          ({ identifier, description }) => ({
            standardIdentifier: identifier,
            description,
          })
        )
 
        const filter = {
          standardIdentifier: {
            $in: practiceStandards.map(
              ({ standardIdentifier }) => standardIdentifier
            ),
          },
        }
        if (!isEmpty(skillIdentifiers)) {
          Object.assign(filter, {
            skillIdentifier: { $in: skillIdentifiers },
          })
        }
        const DifferentiationStandards = await DSM.getByStandardIdentifierList(
          filter
        )
        DifferentiationStandards.forEach(
          ({ reviewStandards = [], challengingStandards = [] }) => {
            recommendations.review.push(...reviewStandards)
            recommendations.challenge.push(...challengingStandards)
          }
        )
        recommendations.practice.push(...practiceStandards)
      }
    }
  }
  recommendations.review = getUniqueRecommendations(recommendations.review)
  recommendations.challenge = getUniqueRecommendations(
    recommendations.challenge
  )
  recommendations.practice = getUniqueRecommendations(recommendations.practice)
  return recommendations
}

export const getAssignmentStudentsByMasteryRange = async ({
  assignmentId,

  min,
  max,
}) => {
  const masteryResult: any = await getAssignmentStudentListWithPerformance({
    assignmentId,

    return masteryResult
  }

      message: 'No students found for give mastery range.',

const checkIfItemsExistForStandards = async (
  collections,
  standards,
  skillIdentifiers
) => {
  if (!standards.length || !collections.length) {
    return false
  }
 
  for (const { _id: collectionId } of collections) {
    for (const { _id: standardId } of standards) {
      const search: any = {
        nInItemIds: [],
        tags: [],
        collectionId,
        standardId,
        skillIdentifiers,
      }
      const [items] = await getAutoSelectSearchItems(1, search)
      if (items.length) {
        return true
      }
    }
  }
  return false
}
 
/**
 * @param {Array} identifiers - Standard identifiers
 * @param {Array} skillIdentifiers - Skill identifiers
 * @param {object} userDetails
 * @param {string} resourceTitle - Title of reference assignment
 * @returns {object}
 */
export const createDynamicTestFromStandardIdentifiers = async ({
  identifiers,
  skillIdentifiers,
  userDetails,
  resourceTitle,
  deliverItemsCount,
}) => {
  const standardIdentifiers = uniq(identifiers)
  const standards = await standardModel.getByCurriculumAndIdentifiers(
    SUPPORTED_CURRICULUM,
    standardIdentifiers
  )
  const sparkMathBank = await new ItemBank().getByOwnerAndName(
    SUPPORTED_COLLECTION,
    SUPPORTED_COLLECTION_NAME,
    { name: 1, type: 1 }
  )
  if (!standards.length || !sparkMathBank) {
    return {
      error: 'Not Found',
      statusCode: 404,
      message:
        'Unable to proceed further. Either standards or collection not found.',
    }
  }
  const collectionsDetails = formCollectionDetails(
    sparkMathBank,
    standards,
    skillIdentifiers
  )
 
  const atleastOneItemExistsInTest = await checkIfItemsExistForStandards(
    collectionsDetails,
    standards,
    skillIdentifiers
  )
 
  if (!atleastOneItemExistsInTest) {
    return {
      error: 'Item Not Found',
      statusCode: 404,
      message: 'No test items could be found in test for chosen standards.',
    }
  }
 
  const test = await createDynamicTestFromCollections({
    userDetails,
    collections: collectionsDetails,
    title: resourceTitle,
    skillIdentifiers,
    deliverItemsCount,
  })
 
  if (!test) {
    return {
      error: 'Internal Server Error',
      statusCode: 500,
      message: 'Unable to create dynamic Test',
    }
  }
 
  return test
}

export const generateStudentRecommendations = async ({
  districtId,
  referenceId,
  referenceName,
  referenceType,
  resourceTitle,
  groupId,
  students,
  testIds = [],
  standardIdentifiers = [],
  skillIdentifiers = [],
  dynamicTest,
  type,
  masteryRange: { min = 0, max = 100 },
  resources,
}) => {
  let recommendations: any
 
  if (testIds.length) {
    const testDetails = await testModel.getByIds(testIds, {
      _id: 0,
      title: 1,
      summary: 1,
    })
    const resourceTitles: Array<string> = []
    const testSummaries: any = []
 
    testDetails.forEach((details) => {
      resourceTitles.push(details.title)
      testSummaries.push(details.summary)
    })
    const testStandards = testSummaries.map((summaries) => {
      return flatMap(summaries.groupSummary, (identifySummary) => {
        return identifySummary.standards.map((standard) => standard.identifier)
      })
    })
    const resourceTitleIdMapper: any = testIds.map(
      (testId: ObjectId, idx: number) => {
        return {
          resourceTitle: resourceTitles[idx],
          testId,
          testStandards: testStandards[idx],
        }
      }
    )
    recommendations = flatMap(resourceTitleIdMapper, (resourceMapper) => {
      return formRecommendations({
        students,
        groupId,
        districtId,
        referenceId,
        referenceName,
        referenceType,
        recommendationType: type,
        masteryRange: { min, max },
        resources,
        recommendedResource: {
          _id: resourceMapper.testId,
          name: resourceMapper.resourceTitle,
          resourceType: 'TEST',
          derivedFrom: 'TESTS',
          metadata: {
            testStandards: resourceMapper.testStandards,
          },
        },
      })
    })
  } else if (!isEmpty(dynamicTest)) {
    recommendations = formRecommendations({
      students,
      groupId,
      districtId,
      referenceId,
      referenceName,
      referenceType,
      recommendationType: type,
      masteryRange: { min, max },
      resources,
      recommendedResource: {
        _id: dynamicTest._id,
        name: resourceTitle,
        resourceType: !isEmpty(dynamicTest) ? 'TEST' : undefined,
        derivedFrom: 'STANDARDS',
        metadata: {
          standardIdentifiers,
          skillIdentifiers,
        },
      },
    })
  }
 
  const insertResult = StudentRecommendation.bulkWrite(
    recommendations,
    'insertOne',
    {}
  )
 
  if (insertResult) {
    return {
      statusCode: 200,
      message: 'Successfully added the recommendations',
    }
  }
  return {
    error: 'Internal Server Error',
    statusCode: 500,
    message: 'Unable to add the recommendations',
  }
}
 
export const getStudentRecommendations = async ({
  userId,
  districtId,
}: {
  [key: string]: string
}) => {
  const recommendations: any = await StudentRecommendation.getByFields({
    userId: new ObjectId(userId),
    districtId: new ObjectId(districtId),
  })
 
  if (isEmpty(recommendations)) {
    return []
  }
  const recommendationIds = recommendations.map((o) => o._id)
  const groupIds = uniq(recommendations.map((o) => `${o.groupId}`))
 
  const recommendedTestActivities = await userTestActivity
    .getByFields(
      {
        groupId: { $in: groupIds },
        studentRecommendationId: { $in: recommendationIds },
      },
      {
        maxScore: 1,
        score: 1,
        status: 1,
        studentRecommendationId: 1,
        testId: 1,
      }
    )
    .read('secondaryPreferred')
    .lean()
 
  for (const recommendation of recommendations) {
    const testActivities = filter(
      recommendedTestActivities,
      (rta) => `${rta.studentRecommendationId}` === `${recommendation._id}`
    )
 
    recommendation.studentTestActivities = testActivities.map((uta) =>
      omit(uta, ['studentRecommendationId'])
    )
  }
  return recommendations
}
 
export const deleteStudentRecommendationsAndUTAs = async (userId) => {
  const userRecommendationIds = await StudentRecommendation.getByFields(
    { userId: new ObjectId(userId) },
    { _id: 1 }
  )
  const ids = userRecommendationIds.map((o) => o._id)
  if (!isEmpty(ids)) {
    await StudentRecommendation.deleteManyByIds(ids)
    await userTestActivity.deleteManyByFields({
      studentRecommendationId: { $in: ids },
    })
  }
}
 
export const getAcceptedRecommendations = async ({
  districtId,
  assignmentId,
  groupId,
}) => {
  const recommendations: any = await StudentRecommendation.getByDistrictAssignmentAndGroup(
    districtId,
    assignmentId,
    groupId
  )
 
  if (!isEmpty(recommendations)) {
    const recommendationIds: any = flatMap(
      recommendations,
      (x) => x.recommendationIds || []
    )
    const recommendedTestActivities: any[] = await userTestActivity
      .getByFields(
        {
          groupId,
          studentRecommendationId: { $in: recommendationIds },
          archived: { $ne: true },
        },
        {
          maxScore: 1,
          score: 1,
          status: 1,
          studentRecommendationId: 1,
          testId: 1,
          userId: 1,
        }
      )
      .read('secondaryPreferred')
      .lean()
 
    for (const recommendation of recommendations) {
      const arr = (recommendation.recommendationIds || []).map((id) => `${id}`)
      const testActivities: any = remove(recommendedTestActivities, (uta) =>
        arr.includes(`${uta.studentRecommendationId}`)
      )
      recommendation.studentTestActivities = testActivities.map((uta) =>
        omit(uta, ['studentRecommendationId'])
      )
    }
  }
  return recommendations
}
