import joi from 'joi'
import {
  releaseGradeType,
  MarkAsDoneType,
  testContentVisibility,
  passwordPolicy,
} from '../const/test'
import { ALL_TEST_TYPES_VALUES } from '../const/testTypes'
import { playerSKinTypes } from '../const/assignment'
import { tagSchema } from '../validators/assignment'
 
export const acceptAssignmentRecommendation = {
  assignmentId: joi.string().required(),
  assignmentSettings: joi.object().keys({
    startDate: joi.number().required(),
    endDate: joi.number().required(),
    openPolicy: joi.string(),
    closePolicy: joi.string(),
    questionPerStandard: joi.number(),
    releaseScore: joi.string().valid(...releaseGradeType),
    testType: joi.string().valid(...ALL_TEST_TYPES_VALUES),
    generateReport: joi.bool(),
    safeBrowser: joi.bool(),
    shuffleQuestions: joi.bool(),
    shuffleAnswers: joi.bool(),
    updateTestActivities: joi.bool(), // dispatch sqs message for updating testActivties
    maxAttempts: joi.number(),
    termId: joi.string(),
    answerOnPaper: joi.bool(),
    sebPassword: joi.string(),
    calcType: joi.string(),
    reportKey: joi.string(),
    markAsDone: joi.string().valid(Object.values(MarkAsDoneType)),
    grades: joi.array(),
    subjects: joi.array(),
    title: joi.string(),
    thumbnail: joi.string(),
    scoringType: joi
      .string()
      .allow(...['ALL_OR_NOTHING', 'PARTIAL_CREDIT', 'ITEM_LEVEL_EVALUATION']),
    penalty: joi.boolean(),
    playlistId: joi.string(),
    playlistModuleId: joi.string(),
    maxAnswerChecks: joi.number(),
    assignmentPassword: joi.string(),
    standardGradingScale: joi.object().keys({
      _id: joi.string(),
      name: joi.string(),
    }),
    performanceBand: joi.object().keys({
      _id: joi.string(),
      name: joi.string(),
    }),
    testContentVisibility: joi
      .string()
      .valid(Object.values(testContentVisibility)),
    passwordPolicy: joi.number().valid(Object.values(passwordPolicy)),
    passwordExpireIn: joi.number(),
    playerSkinType: joi.string().valid(Object.values(playerSKinTypes)),
    showMagnifier: joi.boolean(),
    timedAssignment: joi.bool(),
    pauseAllowed: joi.bool(),
    allowedTime: joi.number(),
    dueDate: joi.number(),
    enableScratchpad: joi.bool(),
    enableSkipAlert: joi.bool(),
    processedUsedItems: joi.array().items(joi.string()),
    v1Attributes: joi.object(),
    schVer: joi.number().optional(),
    tags: joi.array().items(tagSchema).optional(),
    blockNavigationToAnsweredQuestions: joi.boolean(),
    restrictNavigationOutAttemptsThreshold: joi.number().optional(),
    restrictNavigationOut: joi.string().optional().allow(null),
    blockSaveAndContinue: joi.boolean().optional(),
  }),
  groupId: joi.string().required(),
  standardIdentifiers: joi.array().items(joi.string().optional()).min(1),
  skillIdentifiers: joi.array().items(joi.string().optional()),
  type: joi.string().required().valid('REVIEW', 'CHALLENGE', 'PRACTICE'),
  masteryRange: joi.object().keys({
    min: joi.number().required(),
    max: joi.number().required(),
  }),
  resourceTitle: joi.string().required(),
  testIds: joi.when('standardIdentifiers', {
    is: joi.array().length(0),
    then: joi.array().required().min(1),
    otherwise: joi.array().optional(),
  }),
  resources: joi.array().items(joi.object()).optional(),
}
