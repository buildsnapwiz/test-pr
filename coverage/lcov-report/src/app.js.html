/**
 * compress the response based on allowedCompressionPaths
 * @param {*} req
 * @param {*} res
 */
const shouldCompress = (req, res) => {
  const currentPath = req.originalUrl // e.g, /api/report
  // check if the current path is allowed
  const isAllowedPathToCompress = allowedCompressionPaths.some(
    (path) => path instanceof RegExp && path.test(currentPath)
  )
 
  if (!isAllowedPathToCompress) {
    // don't compress response if path is not part of allowed paths
    return false
  }
 
  // fallback to standard filter function
  return compression.filter(req, res)
}
 
app.use(compression({ filter: shouldCompress }))

// log response times in header and catch possible long requests > 30s
app.use(
  responseTime((req, res, time) => {
    const respTimeInSecs = time / 1000
 
    // Start measuring the time. Anything above 30s is problematic
    if (respTimeInSecs > 30) {
      ApiTraceLogger('error', 'POSSIBLE TIMEOUT', req)
    }
    res.setHeader('x-server-ref', req.RequestID)
    res.setHeader('x-server-time', time.toFixed(2))
  })
)
 
// catch if we have explicity timeouts
app.use((req, res, next) => {
  let timeOutSocket
  res.on('timeout', (socket) => {
    timeOutSocket = socket
    ApiTraceLogger('error', 'TIMEOUT', req)
  })
 
  // destroy the socket if we have on our catch
  res.on('finish', () => timeOutSocket && timeOutSocket.destroy())
 
  next()
})

app.use((err, req, res, next) => {
  ApiTraceLogger('error', 'SERVER_ERROR', req, err)
 
  if (Config.sentryDataSourceName) {
    withScope((scope) => {
      const { _id, role, districtId } = req.user || {}
      scope.setUser({ id: _id })
      scope.setTags({
        districtId,
        role,
        kid: getKID(_id, role, districtId),
        requestID: req.RequestID || 'NA',
      })
      if (err instanceof Error || (err && err.message && err.stack)) {
        captureException(err)
      }
    })
  }
  if (res.headersSent) {
    return next(err)
  }
  return res.boom.badImplementation()
})
