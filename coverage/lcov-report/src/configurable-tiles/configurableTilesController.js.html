export const getTestContentRecommendationsCriteria = async (
  user,
  currentDate,
  districtId
) => {
  let standardSets
  let recommendedTiles = []
  if (!_.isEmpty(user.currentStandardSetStandards)) {
    standardSets = Object.keys(user.currentStandardSetStandards)
  } else {
    const { interestedCurriculums } = await getInterestedCurriculums(
      { ...user, currentDistrictId: districtId },
      user._id
    )
    standardSets = interestedCurriculums.map(({ _id }) => _id)
  }
  const district = await new District().getById(districtId).lean()
  const { state = '' } = district?.location || {}
  const emptyStateCacheKeys = standardSets.map((x) => getCacheKey(x, ''))
  const cacheKeys = [
    ...[...standardSets, ''].map((x) => getCacheKey(x, state)),
    ...emptyStateCacheKeys,
  ]
  const result = (await CacheManager.getMultiple(cacheKeys))
    .filter((el) => el)
    .map((el) => JSON.parse(el))
  if (result.length && standardSets.length === cacheKeys.length) {
    recommendedTiles = _.uniqBy(_.flatten(result), (x) => `${x._id}`)
  } else {
    const ConfigurableTiles = new ConfigurableTilesModel()
    recommendedTiles = await ConfigurableTiles.getByFields({
      type: 'RECOMMENDED',
      active: true,
      $and: [
        {
          $or: [{ 'states.0': { $exists: false } }, { states: state }],
        },
        {
          $or: [
            { standardSet: { $exists: false } },
            { standardSet: { $in: standardSets } },
          ],
        },
      ],
    }).lean()
    if (recommendedTiles.length) {
      const dataToCache = {}
      for (const tile of recommendedTiles) {
        const { standardSet = '', states } = tile
        if (states && states.length) {
          for (const st of states) {
            const key = getCacheKey(standardSet, st)
            if (!dataToCache[key]) {
              dataToCache[key] = []
            }
            dataToCache[key].push(tile)
          }
        } else {
          const key = getCacheKey(standardSet, '')
          if (!dataToCache[key]) {
            dataToCache[key] = []
          }
          dataToCache[key].push(tile)
        }
      }
      await CacheManager.setMultipleWithExpiry(
        _.mapValues(dataToCache, (x) => JSON.stringify(x)),
        false
      )
    }
  }
 
  const now = currentDate
  return recommendedTiles.filter((x) => {
    if (x.validFrom && x.validFrom > now) {
      return false
    }
    if (x.validTill && x.validTill < now) {
      return false
    }
    return true
  })
}
 
export const updateCache = async ({ type, state, standardSet }) => {
  const promises = []
  if (['BANNER', 'FEATURED'].includes(type)) {
    promises.push(updateTileVersion())
  }
 
  if (type === 'RECOMMENDED') {
    promises.push(CacheManager.evictKey(getCacheKey(standardSet, state)))
  }
  await Promise.all(promises)
}

export async function getOrderedStandardsAfter(standardIdentifiers) {
  if (!(standardIdentifiers || []).length) return []
 
  const standards = await StandardsModel._getByIds(standardIdentifiers, {
    _id: 1,
    curriculumId: 1,
    position: 1,
  })
    .read('secondaryPreferred')
    .lean()
 
  const result = await StandardsModel.getNextOrderedStandards(standards)
 
  const resultsInOrder = standardIdentifiers.reduce(
    (a, c) => [...a, result[c]],
    []
  )
  return resultsInOrder
}
 
const getDataCacheKey = (
  criteria_id,
  grades,
  subjects,
  standardsSet,
  standards,
  lastUsedStandards
) => {
  const hash = getHmacDigest(
    `${criteria_id}${grades
      .map((x) => x.toLowerCase())
      .join('')}${subjects.join('')}${standardsSet}${standards.join(
      ''
    )}${lastUsedStandards.join('')}`
  )
  return `${CacheableKeys.RECOMMENDED_TESTS_DATA}${hash}`
}
 
/**
 *
 * @param {any} user
 * @param {import('mongodb').ObjectId|null} districtId
 * @param { string|null } role
 */
export async function getTestContentRecommendation(user, districtId, role) {
  const results = []
  const selectedCriteria = await getTestContentRecommendationsCriteria(
    user,
    Date.now(),
    districtId
  )
 
  if (!user.currentStandardSetStandards) {
    user.currentStandardSetStandards = {}
  }
  const interested =
    (await interestedStandardsModel.getUsingOrgIdAndType(
      user._id,
      ORG_TYPE.TEACHER
    )) || {}
  const grades = interested.defaultGrades
  const subjects = interested.defaultSubjects
  if (!(grades && grades.length && subjects && subjects.length)) {
    return []
  }
  for (const criterion of selectedCriteria) {
    const standardSet = criterion.standardSet
      ? parseInt(criterion.standardSet)
      : false
    const currentStandards = standardSet
      ? user.currentStandardSetStandards[standardSet] || []
      : []
 
    const currentStandardIdentifiersSet = new Set(currentStandards)
 
    const nextStandards = await getOrderedStandardsAfter(currentStandards)
 
    const nextStandardsSet = new Set(nextStandards)
 
    const institutionIds = await getCurrentDistrictInstitutionIds(
      user.institutionIds,
      districtId,
      user.districtIds
    )
    const itemBanks = await getItemBanksByRole(
      user._id,
      user.permissions,
      districtId,
      institutionIds,
      role,
      user.districtIds
    )
 
    const expiredPremiumBanks = await getExpiredPremiumBanks({
      districtIds: user.districtIds,
      activeCollectionIds: itemBanks.map((el) => el._id),
    })
 
    const { _id: currentTermId } =
      (await new TermModel().getByDistrictAndDate(districtId, Date.now(), {
        _id: 1,
      })) || {}
 
    const assignments = await AssignmentController.getTeachersAssignments(
      user._id,
      districtId,
      '', // grade
      '', // subject
      currentTermId,
      role,
      '', // tType
      {
        folderId: '', // folderId
        classIds: undefined, // classIds
        tags: [],
        testId: '', // testId
      },
      1 // limit
    )
 
    let lastUsedStandards = []
    let lastUsedTestId = ''
 
    if (assignments.assignments.length) {
      lastUsedTestId = assignments.assignments[0].testId.toString()
      const [test] = await getTestTestItemsAndPassage({
        userId: user._id,
        testId: assignments.assignments[0].testId,
        requestLatest: false,
        editAndRegrade: false,
      })
      if (test.metadata.standards) {
        lastUsedStandards = test.metadata.standards
      }
    }
 
    const cacheKey = getDataCacheKey(
      criterion._id,
      grades,
      subjects,
      standardSet,
      currentStandards,
      lastUsedStandards
    )
    const cachedValue = await CacheManager.get(cacheKey)
    if (cachedValue) {
      results.push({
        criteria_id: criterion._id,
        filters: criterion.config.filters,
        results: JSON.parse(cachedValue),
      })
      Logger.info(
        `cached value used for ${criterion._id}, key:${cacheKey}, userId:${user._id}`
      )
      continue
    }
 
    const allStandards = _.uniq([...currentStandards, ...nextStandards])
 
    const filter = criterion.config.filters && criterion.config.filters[0]
    const collections = filter && filter.collections
 
    const elasticQuery = createTestsSearch(
      user._id,
      districtId,
      institutionIds,
      itemBanks,
      false, // searchedFromPlaylist
      {
        page: 1,
        limit: 100,
        search: {
          filter: 'ENTIRE_LIBRARY',
          standardIds: allStandards,
          grades,
          subject: subjects,
          curriculumId: standardSet,
          forRecommended: true,
          ...(filter && filter.collections && filter.collections.length
            ? { collections }
            : {}),
          tags: criterion.tags || [],
        },
        sort: {
          sortBy: 'popularity',
          sortDir: 'desc',
        },
      },
      false, // hideEngageNy = false
      false, // hasCuratorPermission
      false, // isPublisherAuthor
      false, // isEdulasticCurator
      [], // userFavoritesTestIds
      expiredPremiumBanks
    )
 
    const { data: searchResult } = await _triggerElasticSearch({
      method: 'POST',
      url: `${config.elasticSearch.uri}?type=Tests`,
      data: elasticQuery,
    })
    const standardsBucket = { original: [], next: [] }
    const testList = [...searchResult.hits.hits]
 
    const tests = testList.filter((test) => test._id !== lastUsedTestId)
    const ag = tests[0]
 
    for (const hit of tests) {
      const testStandards = []
      if (hit._source.alignment && hit._source.alignment[0]) {
        for (const domain of hit._source.alignment[0].domains) {
          for (const standard of domain.standards) {
            testStandards.push(standard.id)
          }
        }
      }
      const intersection = testStandards.filter((x) =>
        lastUsedStandards.includes(x)
      )
      const union = [...new Set([...testStandards, ...lastUsedStandards])]
      const popularity = hit.sort[0]
      const recommendationScore =
        (intersection.length / union.length) *
        Math.log10(2 + Math.max(popularity, 0))
      hit.recommendationScore = recommendationScore
      hit.calculationMetrics = {
        // for debugging purposes only
        intersection,
        union,
        lastUsedStandards,
        popularity,
        testStandards,
        lastUsedTestId,
        uid: user._id,
        assignments,
        currentTermId,
        districtId,
      }
    }
 
    tests.sort((a, b) =>
      a.recommendationScore < b.recommendationScore
        ? 1
        : b.recommendationScore < a.recommendationScore
        ? -1
        : 0
    )
 
    const unpickedtests = []
    if (!_.isEmpty(allStandards)) {
      for (const hit of tests) {
        const originalStandardsLessThan3 = standardsBucket.original.length < 3
        const nextStandardLessThan2 = standardsBucket.next.length < 2
        if (
          originalStandardsLessThan3 &&
          hit._source.metadata.standards.find((x) =>
            currentStandardIdentifiersSet.has(x)
          )
        ) {
          standardsBucket.original.push(hit)
        } else if (
          nextStandardLessThan2 &&
          hit._source.metadata.standards.find((x) => nextStandardsSet.has(x))
        ) {
          standardsBucket.next.push(hit)
        } else {
          unpickedtests.push(hit)
        }
      }
    }
 
    let pickedTests = [...standardsBucket.original, ...standardsBucket.next]
    if (pickedTests.length < 5) {
      pickedTests = [
        ...pickedTests,
        ...unpickedtests.slice(0, 5 - pickedTests.length),
      ]
    }
 
    const testResults = pickedTests.length ? pickedTests : tests.slice(0, 5)
    results.push({
      criteria_id: criterion._id,
      filters: criterion.config.filters,
      results: testResults,
    })
 
    if (testResults.length) {
      await CacheManager.setWithExpiry(
        cacheKey,
        JSON.stringify(testResults),
        TTL.SECONDS_IN_DAY * 10
      )
    }
  }
  return results
}
