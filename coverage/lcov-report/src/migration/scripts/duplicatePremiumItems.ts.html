export const duplicatePremiumItems = async () => {
  await initDbConnection()
  try {
    console.log('script initiated...')
    const testItems = await TestItem.getByFields(
      {
        'collections._id': { $in: nonPremiumCollections },
        $where: 'this.collections.length>1',
      },
      {
        v1Id: 0,
        v1Attributes: 0,
        'data.questions.tts': 0,
        'data.questions.rubrics': 0,
        'data.questions.languageFeatures.es.tts': 0,
        cw: 0,
        __v: 0,
      }
    )
    const duplicateItems = {}
    for (let index in testItems) {
      let testItem = testItems[index]
      const user = await User.getById(testItem.createdBy._id)
      const pCollections = []
      const npCollections = []
      for (let collection of testItem.collections) {
        if (nonPremiumCollections.includes(`${collection._id}`)) {
          npCollections.push(collection)
        } else {
          pCollections.push(collection)
        }
      }
      if (pCollections.length) {
        const { _id: originalItemId } = await TestItem.findOneAndUpdate(
          { _id: testItem._id },
          { $set: { collections: pCollections } },
          {}
        )
        const [[duplicateItem]] = await TestItemController.duplicateTestItems(
          [testItem],
          user,
          false
        )
        duplicateItem.collections = npCollections
        await TestItem.bulkInsertItem([duplicateItem])
        duplicateItems[originalItemId] = duplicateItem._id
      }
    }
    console.info('Items update :', duplicateItems)
  } catch (err) {
    console.error('Error while executing script', err)
  }
}
duplicatePremiumItems().then(() => {
  console.info('Script ran successfully..')
  process.exit(0)
})
