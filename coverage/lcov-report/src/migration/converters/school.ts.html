const transformSchool = (
  row,
  v2DistrictId,
  location,
  phone,
  districtName,
  add
) => {
  const schoolObj = {
    v1Id: row.id,
    districtId: v2DistrictId,
    district: {
      _id: v2DistrictId,
      name: districtName,
    },
    updatedAt: commons.getTimeStamp(row.modified_date),
    name: row.name,
    stateId: '',
    schoolNumber: '',
    sisId: '',
    ncesId: row.nces_id,
    lowGrade: '',
    highGrade: '',
    principal: '',
    location,
    mdrNumber: '',
    phone,
    createdAt: commons.getTimeStamp(row.created_date),
    isApproved: Utils.getBoolFromBufferedValue(row.approval_status) ? 1 : 0,
    status: row.status === 1 ? 0 : 1,
    // in V1 ACTIVE_STATUS = 0 , INACTIVE_STATUS = 1
  }
 
  let v1Attributes: any = {
    v1Code: row.code,
    v1CreatedBy: row.created_by,
    v1Description: row.description,
    v1DisplayName: row.display_name,
    v1ModifiedBy: row.modified_by,
    v1CustomerId: row.customer_id,
    v1Type: row.type,
    v1ShippingAddressId: row.shipping_address_id,
    v1Position: row.position,
    v1InstitutionType: row.institution_type,
    v1NameCopy: row.name_copy,
  }
 
  if (!_.isEmpty(add) && !_.isEmpty(add[0])) {
    Object.assign(v1Attributes, {
      v1CountryName: add[0].country_name,
    })
  }
 
  v1Attributes = Utils.removeNullOrUndefinedFieldsWithKey(v1Attributes)
  Object.assign(schoolObj, {
    v1Attributes,
  })
 
  if (!_.isNil(row.clever_id)) {
    Object.assign(schoolObj, {
      cleverId: row.clever_id,
    })
  }
 
  return schoolObj
}
 
export const migrateSchool = async (
  v2DistrictId,
  id,
  doOverride,
  getClasses = true,
  migrateAssociatedDistrict = false,
  districtName = '',
  migrateSA = false,
  getSubscriptions = false,
  v1SchoolYearId = undefined,
  isHomeSchool = false,
  ignoreDataMigrationLog = false,
  overrideQuestions = false,
  migrationVersion = 0,
  overridePassage = false,
  restrictedMigration = false,
  archiveMigration = false,
  archivedConnection = null
) => {
  let rows = await knex.select().table('t_institution').where({ id })
 
  if (!rows.length) {
    throw new Error('School not found')
  }
  const v1School = rows[0]
  const add = await knex
    .select()
    .table('t_address')
    .where({ id: v1School.shipping_address_id })
 
  let location
  let phone
  if (!_.isEmpty(add)) {
    location = {
      city: add[0].city,
      state: add[0].state,
      zip: add[0].zip,
      address: add[0].street,
      country: add[0].country_code,
    }
    phone = add[0].phone
  }
  if (!v2DistrictId && migrateAssociatedDistrict) {
    const response = await migrateDistrict(
      v1School.institution_district_id,
      false,
      false,
      false,
      false,
      false,
      false,
      false,
      undefined,
      false,
      0,
      false,
      false,
      archiveMigration,
      archivedConnection
    )
    const { district } = response
    v2DistrictId = district._id
    districtName = district.name
  }
 
  const school = transformSchool(
    v1School,
    v2DistrictId,
    location,
    phone,
    districtName,
    add
  )
  if (isHomeSchool) {
    // we don't want homeschool to be searchable
    Object.assign(school, { status: Status.DISABLE })
  }
  const queryFilter = {
    v1Id: id,
  }
  if (!restrictedMigration && !archiveMigration) {
    Object.assign(queryFilter, {
      districtId: v2DistrictId,
      updatedAt: { $gt: school.updatedAt || 0 },
    })
  }
  let v2School = (await schoolModel.School.findOne(
    queryFilter
  ).lean()) as SchoolType
  // Moved few schools to different district.
  // throw an error to ristrict further migration
  // otherwise it create new school record in current district.
  if (restrictedMigration) {
    if (v2School && `${v2School.districtId}` !== `${v2DistrictId}`) {
      throw new Error(`school ${id} now part of different district`)
    }
  }
  if (!v2School || doOverride) {
    try {
      if (restrictedMigration) school.status = Status.DISABLE
 
      v2School = (await new SchoolModel(
        archiveMigration ? archivedConnection : null
      ).School.findOneAndUpdate(
        { districtId: v2DistrictId, v1Id: id },
        // @ts-ignore
        school,
        {
          upsert: true,
          new: true,
          timestamps: false,
        }
      ).lean()) as SchoolType
    } catch (duex) {
      if (duex.message.includes('duplicate key error')) {
        Logger.info(
          `School duplicate encountered for v1Id ${id} for District ${v2DistrictId}`
        )
        v2School = (await new SchoolModel(
          archiveMigration ? archivedConnection : null
        ).School.findOne({
          districtId: v2DistrictId,
          v1Id: id,
        }).lean()) as SchoolType
      } else {
        throw duex
      }
    }
 
    try {
      await migratePermissions(
        id,
        v2DistrictId,
        {},
        doOverride,
        restrictedMigration,
        archiveMigration,
        archivedConnection
      )
    } catch (e) {
      console.log(`Premium Content Permission failed for school ${id}`, e)
    }
  }
 
  if (migrateSA) {
    await migrateSchoolAdmins(
      v2DistrictId,
      id,
      v1SchoolYearId,
      'ROLE_SA_ADMIN',
      v2School._id,
      v1School.institution_district_id,
      doOverride,
      overrideQuestions,
      migrationVersion,
      overridePassage,
      restrictedMigration,
      archiveMigration,
      archivedConnection
    )
  }
  if (getSubscriptions) {
    orgSubscriptionMessage.MessageBody = JSON.stringify({
      subscriptionType: 'school',
      v1DistrictId: v1School.institution_district_id,
      v2DistrictId,
      v1SchoolId: v1School.id,
      v2SchoolId: v2School._id,
      v1UserId: '',
      v2UserId: '',
      override: doOverride,
      restrictedMigration,
      messageType: messageTypes.MIGRATE_ORG_SUBSCRIPTION,
      archiveMigration,
    })
    await sqs.send(orgSubscriptionMessage)
    // uncomment below to run in local
    // await migrateOrgSubscription(
    //   'school',
    //   v1School.institution_district_id,
    //   v2DistrictId,
    //   v1School.id,
    //   v2School._id,
    //   '',
    //   '',
    //   doOverride,
    //   restrictedMigration,
    //   archiveMigration,
    //   archivedConnection,
    // )
  }
  if (getClasses) {
    const classQuery = knex
      .select('class_section_id', 'primary_teacher_id')
      .table('d_class_section')
      .where({ school_id: id })
    if (v1SchoolYearId) {
      classQuery.where({ school_year_id: v1SchoolYearId })
    }
    rows = await classQuery
    if (!rows.length) {
      console.log(`Class Sections not found for school ${id}`)
    }
  }
  if (!ignoreDataMigrationLog) {
    await new DataMigrationLog(
      archiveMigration ? archivedConnection : null
    ).update(
      { v1DistrictId: v1School.institution_district_id, active: 1 },
      { $inc: { schoolProcessedCount: 1 } }
    )
  }
  return {
    rows,
    school: {
      _id: v2School._id,
      name: v2School.name,
      districtId: v2School.districtId,
    },
  }
}
 
export const getSchool = async (
  schoolId,
  v2DistrictId = undefined,
  isHomeSchool = false,
  archiveMigration = false,
  archivedConnection = null
) => {
  let school = await new SchoolModel().School.findOne(
    {
      districtId: v2DistrictId,
      v1Id: schoolId,
    },
    { districtId: 1 }
  ).lean()
  if (archiveMigration && !school) {
    school = await new SchoolModel(archivedConnection).School.findOne(
      {
        districtId: v2DistrictId,
        v1Id: schoolId,
      },
      { districtId: 1 }
    ).lean()
  }
  if (_.isEmpty(school)) {
    // v2DistrictId, id, doOverride, getClasses, migrateAssociatedDistrict, districtName,
    // migrateSA, getSubscriptions, v1SchoolYearId, isHomeSchool
    // the defaults are sent in the below way because they are not need in this flow
    const response = await migrateSchool(
      v2DistrictId,
      schoolId,
      false,
      false,
      true,
      undefined,
      false,
      false,
      undefined,
      isHomeSchool,
      true,
      false,
      0,
      false,
      false,
      archiveMigration,
      archivedConnection
    )
    // @ts-ignore
    ;({ school } = response)
  }
  return school
}
