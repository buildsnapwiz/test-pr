const cleanSpacesBeforeNewLine = (str) => {
  let iBr = str.indexOf('<br')
  while (iBr !== -1) {
    const start = str.indexOf('>', iBr)
    const left = str.slice(0, start + 1)
    let right = str.slice(start + 1)
    str = str.replace(/^\s+/, '')
    str = str.replace(/^(&nbsp;)+/, '')
    str = left + right
    iBr = str.indexOf('<br', iBr + 1)
  }
  return str
}
 
export const gpConverter = async (
  v1Question,
  alignment,
  user,
  v1DistrictId,
  archiveMigration = false,
  archivedConnection = null
) => {
  const jsonData = JSON.parse(v1Question.json_data)
 
  const [testItem, title] = await Promise.all([
    Question.getTestItemLevelAttributes(
      user,
      v1Question,
      v1DistrictId,
      archiveMigration,
      archivedConnection
    ),
    convertTitleMathMLToLaTeX(jsonData.title, v1Question.id, v1DistrictId),
  ])
  const correctAnsJSON = JSON.parse(v1Question.correct_answer_json)
 
  const convertedJson = graphPlotterTemplate(jsonData.id, alignment, testItem)
  const [entity] = convertedJson.data.questions
  const [widget] = convertedJson.rows[0].widgets
  entity.stimulus = cleanSpacesBeforeNewLine(
    Utils.sanitizeRedactorTagsAndWrapWithPara(
      title,
      v1Question.content_ownership
    )
  )
  entity.stimulus = Utils.addFormulaMargin(entity.stimulus)
 
  const { graphProperties } = jsonData
  Object.assign(entity.uiStyle, {
    xAxisLabel: graphProperties.labelX,
    yAxisLabel: graphProperties.labelY,
    stepSize: graphProperties.scaleY,
    fractionFormat: graphProperties.yAxisFormat,
    yAxisMax: parseFloat(graphProperties.maxY),
    yAxisMin: parseFloat(graphProperties.minY),
  })
  entity.chart_data.name = ''
 
  if (graphProperties.chartType === 'column') {
    entity.uiStyle.chartType = 'bar'
    entity.title = 'Bar chart'
    entity.type = 'bar'
    widget.title = 'Bar chart'
    widget.type = 'bar'
  } else {
    entity.uiStyle.chartType = 'line'
    entity.title = 'Line chart'
    entity.type = 'line'
    widget.title = 'Line chart'
    widget.type = 'line'
  }
 
  if (graphProperties.snapsPerLine) {
    const snapTo = graphProperties.scaleY / graphProperties.snapsPerLine
    entity.uiStyle.snapTo = parseFloat(snapTo.toFixed(4))
  } else {
    entity.uiStyle.snapTo = entity.uiStyle.stepSize
  }
 
  const chartData: Array<{ x: string | number; y: string | number }> = []
  const correctAnswerValues: Array<{
    x: string | number
    y: string | number
  }> = []
 
  graphProperties.categories.forEach((category) => {
    let startValue = parseFloat(graphProperties.startingValues[category.id])
    if (isNaN(startValue) || startValue > entity.uiStyle.yAxisMax) {
      startValue = entity.uiStyle.yAxisMax
    }
 
    let correctAnswerValue = parseFloat(correctAnsJSON[category.id])
    if (
      isNaN(correctAnswerValue) ||
      correctAnswerValue > entity.uiStyle.yAxisMax
    ) {
      correctAnswerValue = entity.uiStyle.yAxisMax
    }
 
    chartData.push({
      x: category.value,
      y: startValue,
    })
    correctAnswerValues.push({
      x: category.value,
      y: correctAnswerValue,
    })
  })
 
  entity.chart_data.data = chartData
  entity.validation.validResponse.value = correctAnswerValues
 
  return convertedJson
}
 
const converter = async (
  v1Id,
  alignment,
  v1DistrictId?,
  v1QuestionObj?,
  archiveMigration = false,
  archivedConnection = null
) => {
  if (!v1QuestionObj) {
    const v1Questions = await knex
      .select()
      .table('t_questions')
      .where({ id: v1Id })
    if (!v1Questions.length) {
      throw new Error('Resource not found')
    }
    v1QuestionObj = v1Questions[0]
  }
 
  const user = await Question.getAuthor(
    v1QuestionObj.created_by,
    archiveMigration,
    archivedConnection
  )
  const convertedQuestion = await gpConverter(
    v1QuestionObj,
    alignment,
    user,
    v1DistrictId,
    archiveMigration,
    archivedConnection
  )
  return convertedQuestion
}
 
const userResponseTransformer = (uqh, v1Question, v2Question) => {
  const { skip, ur, er } = uqh
  if (!skip && !isEmpty(ur) && !isEmpty(er)) {
    const {
      graphProperties: { categories },
    } = JSON.parse(v1Question.json_data)
    const userResponseParsed = JSON.parse(ur)
    const { evaluation: evaluatedResponseParsed } = JSON.parse(er)
    const userResponse: {
      x: string | number
      y: string | number
    }[] = []
    const evaluation: boolean[] = []
    categories.forEach((ca) => {
      evaluation.push(evaluatedResponseParsed[ca.id])
      userResponse.push({
        x: ca.value,
        y: userResponseParsed[ca.id],
      })
    })
    return [evaluation, userResponse]
  }
  return [[], []]
}
