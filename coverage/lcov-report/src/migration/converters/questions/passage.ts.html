const options = { upsert: true, new: true, timestamps: false }

const getPassage = (v1questionSetDetails, authors, createdBy, collections) => {
  const mVersion = 0
  const contentJSON = JSON.parse(v1questionSetDetails.content)
  const tts = !_.isEmpty(v1questionSetDetails.qsd_attributes)
    ? JSON.parse(v1questionSetDetails.qsd_attributes)
    : undefined
  const data: any = []
  const widgets: Array<object> = []
  const tabs: Array<string> = []
  contentJSON.forEach((tab, index) => {
    const entityId = commons.getEntityId(index)
    data.push({
      id: `${entityId}`,
      title: 'Passage',
      type: 'passage',
      heading: `${v1questionSetDetails.instructions}`,
      math_renderer: '',
      content: `${sanitizeRedactorTagsAndWrapWithPara(tab.tabContent, '')}`,
      contentsTitle: `${tab.tabTitle}`,
      variable: {
        variables: {},
      },
    })
    widgets.push({
      widgetType: 'resource',
      type: 'passage',
      title: 'Passage',
      reference: `${entityId}`,
      tabIndex: index,
    })
    // create tabs only when there are more than 1 tabs
    if (contentJSON.length > 1) {
      tabs.push(`${tab.tabTitle}`)
    }
  })
  const passage: PassageType = {
    v1Id: v1questionSetDetails.id,
    data,
    status: 'draft',
    testItems: [],
    structure: {
      tabs,
      widgets,
      dimension: '50%',
      flowLayout: false,
      content: '',
    },
    tts,
    createdBy,
    authors,
    createdAt: commons.getTimeStamp(v1questionSetDetails.created_date),
    updatedAt: commons.getTimeStamp(v1questionSetDetails.modified_date),
    collections,
    mVersion,
  }
  return passage
}
 
const getPassageTestItems = async (
  v1QuestionSetDetailsId,
  passageId,
  passageContent,
  v1Questions,
  v1DistrictId,
  sharedType,
  collections,
  authors,
  createdBy,
  questionMap,
  questionSkillMap,
  migrationVersion,
  archiveMigration = false,
  archivedConnection = null
) => {
  const v2TestItems: any = []
  const questionScoreMap = await Question.getQuestionScoreMap(
    v1Questions.map((q) => q.id)
  )
 
  for (const v1Question of v1Questions) {
    let questionScore
    if (
      !!questionScoreMap[v1Question.id] &&
      !!questionScoreMap[v1Question.id].length
    ) {
      const [firstQuestionSetItem] = questionScoreMap[v1Question.id]
      const { points: maxScore } = firstQuestionSetItem
      questionScore = _.isNumber(maxScore) ? maxScore : 1
    } else {
      questionScore = await Question.getMaxScore(v1Question.id)
    }
    const questionObj =
      !!questionMap && !!questionMap[v1Question.id]
        ? questionMap[v1Question.id][0]
        : null
    const questionSkills = !!questionSkillMap
      ? questionSkillMap[v1Question.id]
      : null
    let v2Question = await convertQuestion({
      v1Id: v1Question.id,
      qType: v1Question.question_type,
      v1DistrictId,
      questionScore,
      questionObj,
      questionSkills,
      migrationVersion,
      archiveMigration,
      archivedConnection,
    })
    Object.assign(v2Question, {
      v1Id: `${v1Question.id}_${v1QuestionSetDetailsId}`,
    })
 
    _.set(v2Question, 'passageId', passageId)
    _.set(v2Question, 'passageContent', passageContent)
    _.set(v2Question, 'multipartItem', true)
    _.set(v2Question, 'canAddMultipleItems', true)
    _.set(v2Question, 'rows[0].dimension', '50%')
    _.set(v2Question, 'collections', collections) // setting collections based on test content ownership.
    _.set(v2Question, 'authors', authors) //Passage authors will be authors for passage questions.
    _.set(v2Question, 'createdBy', createdBy) // Passage creator will be creator for passage questions.
 
    if (sharedType) {
      _.set(v2Question, 'sharedType', sharedType)
    }
 
    if (v2Question.v1Attributes) {
      v2Question.v1Attributes = removeNullOrUndefinedFieldsWithKey(
        v2Question.v1Attributes
      )
    }
    // look up by composite id to support item duplication
    let testItem = await testItemModel.TestItem.findOne({
      v1Id: `${v1Question.id}_${v1QuestionSetDetailsId}`,
      $or: [
        { updatedAt: { $gt: v2Question.updatedAt || '0' } },
        { mVersion: { $gt: migrationVersion } },
      ],
    })
    if (!testItem) {
      try {
        Logger.info(
          `Overriding testItem ${v1Question.id}_${v1QuestionSetDetailsId} for passage ${passageId}`
        )
        const _id = new ObjectID()
        v2Question.mVersion = migrationVersion
        testItem = await new TestItem(
          archiveMigration ? archivedConnection : null
        ).TestItem.findOneAndUpdate(
          { v1Id: `${v1Question.id}_${v1QuestionSetDetailsId}` },
          {
            $set: v2Question,
            $setOnInsert: {
              _id,
              versionId: _id,
              previousTestItemId: _id,
            },
          },
          options
        )
      } catch (duex) {
        if (duex.message.includes('duplicate key error')) {
          Logger.info(
            `Test Item duplicate encountered for v1QuestionId ${v1Question.id} of QuestionSet ${v1QuestionSetDetailsId}`
          )
          testItem = await new TestItem(
            archiveMigration ? archivedConnection : null
          ).TestItem.findOne({
            v1Id: `${v1Question.id}_${v1QuestionSetDetailsId}`,
          })
        } else {
          throw duex
        }
      }
      if (testItem && v2Question.collections) {
        await Question.migrateContentOwnership(
          v2Question.collections,
          v2Question.createdBy,
          BUCKET_CONTENT_ITEMS_TYPE.TESTITEM,
          testItem._id,
          null,
          testItem.createdAt,
          testItem.updatedAt,
          archiveMigration,
          archivedConnection
        )
      }
    } else {
      let message = `override skipped due to modified content in v2 for question ${testItem.v1Id}`
      Logger.info(message)
      new FailedJobs(archiveMigration ? archivedConnection : null).create({
        v1DistrictId,
        jobName: 'override-test-item',
        message: JSON.stringify({
          v1QuestionId: testItem.v1Id,
          v2UpdatedAt: testItem.updatedAt,
        }),
        errorMessage: message,
      })
    }
    v2TestItems.push(testItem)
  }
  return v2TestItems
}
 
const savePassage = async (
  v1Id,
  passage,
  sharedWith,
  sharedType,
  migrationVersion,
  archiveMigration = false,
  archivedConnection = null
) => {
  let newPassage
  try {
    Logger.info(`Overriding passage ${v1Id}`)
    newPassage = (await new Passage(
      archiveMigration ? archivedConnection : null
    ).Passage.findOneAndUpdate(
      { v1Id: v1Id },
      _.omitBy(
        Object.assign(passage, {
          sharedWith,
          sharedType,
          mVersion: migrationVersion,
        }),
        isNullOrUndefined
      ),
      options
    )) as PassageModelType
    return newPassage
  } catch (duex) {
    if (duex.message.includes('duplicate key error')) {
      Logger.info(`Passage duplicate encountered for v1Id ${v1Id}`)
      newPassage = await new Passage(
        archiveMigration ? archivedConnection : null
      ).Passage.findOne({
        v1Id,
      }).lean()
      return newPassage
    }
    throw duex
  }
}
 
const savePassageNew = async (
  passageId,
  v2TestItemIds,
  status,
  archiveMigration = false,
  archivedConnection = null
) => {
  let newPassage
  try {
    newPassage = (await new Passage(
      archiveMigration ? archivedConnection : null
    ).Passage.findOneAndUpdate(
      { _id: passageId },
      { $set: { testItems: [...v2TestItemIds] }, status },
      { new: true, timestamps: false }
    )) as PassageType
    return newPassage
  } catch (duex) {
    if (duex.message.includes('duplicate key error')) {
      Logger.info(`Passage duplicate encountered for _id ${passageId}`)
      newPassage = await new Passage(
        archiveMigration ? archivedConnection : null
      ).Passage.findOne({
        _id: passageId,
      }).lean()
      return newPassage
    }
    throw duex
  }
}
async function passageConverter(
  v1QuestionSets: any,
  v1DistrictId: number,
  questionSetDetails: any,
  sharedWith,
  sharedType,
  collections,
  questionMap,
  questionSkillMap,
  migrationVersion,
  archiveMigration = false,
  archivedConnection = null
) {
  if (!questionSetDetails) {
    const questionSetDetailsList = await knex
      .select()
      .table('t_question_set_details')
      .where('id', v1QuestionSets[0].question_set_details_id)
    if (!questionSetDetailsList.length) {
      throw new Error(
        `Question set details not found for id ${v1QuestionSets[0].question_set_details_id}`
      )
    } else {
      ;[questionSetDetails] = questionSetDetailsList
    }
  }
 
  const user = await Question.getAuthor(
    questionSetDetails.created_by,
    archiveMigration,
    archivedConnection
  )
 
  const questions = await knex({ tqs: 't_question_set' })
    .distinct('tq.id', 'tq.question_type')
    .joinRaw(',`t_question_set_item` as `tqsi`')
    .whereRaw('tqs.id = tqsi.question_set_id')
    .joinRaw(',`t_questions` as `tq`')
    .whereRaw('tq.id = tqsi.question_id')
    .whereIn(
      'tqs.id',
      v1QuestionSets.map((el) => el.id)
    )
  if (!questions.length) {
    throw new Error(
      `Questions not found for qsetId ${v1QuestionSets.map((el) => el.id)}`
    )
  }
 
  const authors = Question.getAuthors(user)
  const createdBy = Question.getCreatedBy(user)
 
  let passage = getPassage(questionSetDetails, authors, createdBy, collections)
 
  // if content is updated in v2, should not re-migrate
  let v2passage = (await new Passage().Passage.findOne({
    v1Id: questionSetDetails.id,
    $or: [
      { updatedAt: { $gt: passage.updatedAt || 0 } },
      { mVersion: { $gt: migrationVersion } },
    ],
  })) as PassageModelType
 
  if (!v2passage && archiveMigration) {
    v2passage = (await new Passage(archivedConnection).Passage.findOne({
      v1Id: questionSetDetails.id,
      $or: [
        { updatedAt: { $gt: passage.updatedAt || 0 } },
        { mVersion: { $gt: migrationVersion } },
      ],
    })) as PassageModelType
  }
  const { _id: passageId, data } =
    v2passage ||
    (await savePassage(
      questionSetDetails.id,
      passage,
      sharedWith,
      sharedType,
      migrationVersion,
      archiveMigration,
      archivedConnection
    ))
 
  const passageContent = getPassageContent(data)
 
  const v2TestItems = await getPassageTestItems(
    questionSetDetails.id,
    passageId,
    passageContent,
    questions,
    v1DistrictId,
    sharedType,
    collections,
    authors,
    createdBy,
    questionMap,
    questionSkillMap,
    migrationVersion,
    archiveMigration,
    archivedConnection
  )
  const v2TestItemIds = v2TestItems.map((testItem) => testItem._id)
  const { status } = v2TestItems[0]
  passage =
    v2passage ||
    (await savePassageNew(
      passageId,
      v2TestItemIds,
      status,
      archiveMigration,
      archivedConnection
    ))
 
  return passage
}
 
export const converter = async (
  v1Id,
  v1DistrictId,
  sharedWith,
  sharedType,
  collections,
  questionMap,
  questionSkillMap,
  migrationVersion,
  archiveMigration = false,
  archivedConnection = null
) => {
  const questionSetDetails = await knex
    .select()
    .table('t_question_set_details')
    .where({ id: v1Id })
  if (!questionSetDetails.length) {
    throw new Error('Resource not found')
  }
 
  // get all the question sets linked to questionSetDetailId
  let questionSets = await knex({ tqs: 't_question_set', ta: 't_assessment' })
    .select('tqs.*')
    .whereRaw(
      knex.raw(`tqs.question_set_details_id = ${questionSetDetails[0].id}
      and tqs.assessment_id = ta.id
      and (
        select count(tqsi_4.question_set_id)
      from
        t_question_set_item tqsi_4
      where
        tqs.id = tqsi_4.question_set_id) > 0 
        `)
    )
 
  if (!questionSets.length) {
    throw new Error(
      `QuestionSets not found for qsetDtl ${questionSetDetails[0].id}`
    )
  }
 
  return passageConverter(
    questionSets,
    v1DistrictId,
    questionSetDetails[0], //order will be by id and least value of id will give the original passage content so picking first
    sharedWith,
    sharedType,
    collections,
    questionMap,
    questionSkillMap,
    migrationVersion,
    archiveMigration,
    archivedConnection
  )
}
