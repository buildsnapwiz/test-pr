export const processTags = async (
  tags: string,
  districtId?: string | ObjectID,
  tagType?: string,
  archiveMigration = false,
  archivedConnection = null
) => {
  if (_.isEmpty(tags)) {
    return []
  }
  const tagsToSave: Array<{
    tagName: string
    districtId?: ObjectID
    tagType?: string
  }> = []
  const tagList = JSON.parse(tags) as Array<string>
  if (_.isEmpty(tagList)) {
    return tagList
  }
  tagList.forEach((tagName) => {
    const tagToSave: {
      tagName: string
      districtId?: ObjectID
      tagType?: string
    } = { tagName }
    if (!_.isEmpty(districtId)) {
      Object.assign(tagToSave, { districtId: new ObjectID(`${districtId}`) })
    }
    if (!_.isEmpty(tagType)) {
      Object.assign(tagToSave, { tagType })
    }
    tagsToSave.push(tagToSave)
  })
  if (_.isEmpty(tagsToSave)) {
    return tags
  }
  const savedTags = (await Promise.all(
    tagsToSave.map(async (tagObj) => {
      let existingTag
      if (archiveMigration) {
        existingTag = await new TagsModel().Tag.findOne(tagObj).lean()
        if (_.isEmpty(existingTag)) {
          return new TagsModel(archivedConnection).Tag.findOneAndUpdate(
            tagObj,
            tagObj,
            {
              new: true,
              upsert: true,
            }
          ).lean()
        }
        return existingTag
      } else {
        return new TagsModel().Tag.findOneAndUpdate(tagObj, tagObj, {
          new: true,
          upsert: true,
        }).lean()
      }
    })
  )) as TagsTyping[]
  return savedTags.map(({ _id, tagName }) => ({ _id, tagName }))
}

const getComputedDimensions = (naturalWidth, naturalHeight) => {
  let width
  let height
  if (naturalHeight > maxHeight || naturalWidth > maxWidth) {
    const fitHeight = Math.floor(maxWidth * (naturalHeight / naturalWidth))
    const fitWidth = Math.floor(maxHeight * (naturalWidth / naturalHeight))
    if (fitWidth > maxWidth) {
      width = maxWidth
      height = fitHeight
    } else {
      height = maxHeight
      width = fitWidth
    }
  } else {
    width = naturalWidth
    height = naturalHeight
  }
  return { _imageWidth: width, _imageHeight: height }
}

export const setImageAttributesFromBGCSS = async (
  backgroundCSS,
  entity,
  isDefaultStructure = true
) => {
  const elementWithBGStyle = $('<span/>', { attr: { style: backgroundCSS } })
  let img = elementWithBGStyle.css('background-image')
  img = img.split('(')[1]
  img = img && img.replace(/\)/, '')
  if (img) {
    let imageWidth = parseInt(elementWithBGStyle.css('width'))
    let imageHeight = parseInt(elementWithBGStyle.css('height'))
    const imageOptions = {
      x: parseInt(elementWithBGStyle.css('left')),
      y: parseInt(elementWithBGStyle.css('top')),
    }
 
    const {
      height: actualHeight,
      width: actualWidth,
      error,
    } = await getImageDimension(img)
 
    if (!error) {
      const aspectRatioDiff =
        actualHeight / actualWidth - imageHeight / imageWidth
      if (aspectRatioDiff < 0) {
        imageHeight = (imageWidth * actualHeight) / actualWidth
      }
      if (aspectRatioDiff > 0) {
        imageWidth = (actualWidth * imageHeight) / actualHeight
      }
    }
    const { _imageWidth, _imageHeight } = getComputedDimensions(
      imageWidth,
      imageHeight
    )
    if (isDefaultStructure) {
      Object.assign(entity, {
        imageUrl: img,
        imageOriginalWidth: imageWidth,
        imageWidth: _imageWidth,
        imageOptions,
        imageHeight: _imageHeight,
        imageOriginalHeight: imageHeight,
      })
    } else {
      Object.assign(imageOptions, { width: imageWidth, height: imageHeight })
      Object.assign(entity, {
        imageOriginalWidth: imageWidth,
        imageOriginalHeight: imageHeight,
        imageUrl: img,
        imageOptions,
      })
    }
  }
}

export const getSharing = async (
  { sharedWithIds },
  archiveMigration = false,
  archivedConnection = null
) => {
  let sharedType
  const sharedWith: Array<{
    _id: ObjectID
    name: string
  }> = []
  const userModel = new UserModel()
  if (!_.isEmpty(sharedWithIds)) {
    const ids = sharedWithIds.split(',')
    for (const id of ids) {
      const [v1ShareLevel, v1SharedId] = id.split('_')
      sharedType = Constants.sharingLevel[v1ShareLevel]
      switch (sharedType) {
        case 'DISTRICT': {
          try {
            const district = await getDistrict(
              v1SharedId,
              false,
              archiveMigration,
              archivedConnection
            )
            sharedWith.push({
              _id: district._id,
              name: district.name,
            })
          } catch (e) {
            sharedType = ''
            Logger.info(`district not Found for v1SharedId :  ${sharedWithIds}`)
          }
          break
        }
        case 'SCHOOL': {
          try {
            const school = (await getSchool(
              v1SharedId,
              undefined,
              false,
              archiveMigration,
              archivedConnection
            )) as SchoolType
            sharedWith.push({
              _id: school._id,
              name: school.name,
            })
          } catch (e) {
            sharedType = ''
            Logger.info(`School not Found for v1SharedId :  ${sharedWithIds}`)
          }
          break
        }
        case 'INDIVIDUAL': {
          let user = (await userModel.User.findOne({
            v1Id: v1SharedId,
          }).lean()) as UserType
          if (!user) {
            user = (await migrateUser(
              v1SharedId,
              false,
              false,
              archiveMigration,
              archivedConnection
            )) as UserType
          }
          sharedWith.push({
            _id: user._id,
            name: _.without(
              [user.firstName, user.lastName],
              undefined,
              null,
              ''
            ).join(' '),
          })
          sharedType = undefined
          break
        }
        default:
          break
      }
    }
  }
  return { sharedWith, sharedType }
}

export const cleanMathJaxMarkupWithProperStructure = (
  htmlText,
  className = '.swformula'
) => {
  htmlText = $.trim(htmlText)
  if (htmlText && htmlText.length == 0) {
    return htmlText
  }
  const content = $('<div>').append(htmlText)
 
  $(content)
    .find('mspace')
    .each(function (this: any) {
      if (this.hasAttribute('newline')) {
        // some time mspace tag is coming as : <mspace linebreak=\"\" newline\"\"=\"\"></mspace>, it should be <mspace linebreak="newline" ></mspace>
        $(this).remove()
      }
    })
  $(content)
    .find('maction')
    .each(function (this: any) {
      if (this.hasAttribute('argument')) {
        // some time maction tag is coming as : <maction actiontype=\"\" argument\"\"=\"\"></maction>, it should be <maction actiontype="argument" ></maction>
        $(this).remove()
      }
    })
  if (content.find('#MathJax_Message').length > 0) {
    content.find('#MathJax_Message').remove()
  }
 
  content.find(className).each(function (this: any) {
    $(this).removeAttr('style')
    if (
      $(this).find('.MathJax').length > 0 ||
      $(this).find('.MathJax_MathML').length > 0 ||
      $(this).find('.MathJax_Preview').length ||
      $(this).find('#MathJax_Message').length > 0
    ) {
      $(this).find('.MathJax').remove()
      $(this).find('#MathJax_Message').remove()
      $(this).find('.MathJax_MathML').remove()
      const mathMlContent = $(this).find('script').html()
      if (mathMlContent && mathMlContent.length > 0) {
        $(this).find('script').remove()
        $(this).html(mathMlContent)
      }
    }
  })
  return content.html()
}

export const sanitizeRedactorTags = (inputString) => {
  if (inputString !== null && inputString !== undefined) {
    inputString =
      typeof inputString === 'string' ? inputString : inputString.toString()
  } else if (!inputString) {
    inputString = ''
  }
  inputString = inputString
    .replace(/<inline/gi, '<span')
    .replace(/inline>/gi, 'span>')
    .replace(/<div style=\"clear: both;\"><\/div>/g, '<br>')
    .replace(/<v>/gi, '')
    .replace(/<\/v>/gi, '')
 
  let tempDiv = $('<div/>').append(inputString)
  if (tempDiv.find('audio').length) {
    tempDiv.find('audio source').each(function (idx, obj) {
      $(obj).parent().attr('src', $(obj).attr('src')).attr('controls', '')
      $(obj).remove()
    })
  }
 
  if (tempDiv.find('img').length) {
    tempDiv.find('img').each(function (idx, obj) {
      const sanitizedImg = sanitizeHtml(obj.outerHTML, sanitizeDefaultConfig)
      $(obj).replaceWith(
        $(sanitizedImg)
          .css('vertical-align', 'middle')
          .addClass('fr-fic fr-dii')
      )
    })
  }
  if (tempDiv.find('a').length) {
    tempDiv.find('a').attr('target', '_blank')
  }
 
  if (tempDiv.find('klm').length) {
    tempDiv.find('klm').each(function (i, elem) {
      $(elem).replaceWith($(elem).html())
    })
  }

const computeBoxWidth = (displayWidth) => {
  const v2DefaultBoxWidth = 140
  displayWidth = parseInt(displayWidth)
  if (isNaN(displayWidth)) {
    return v2DefaultBoxWidth
  }
  // Using logic similar to v1
  const maxV1WidthFactor = 120
  const v1MultiplicationFactor = 7
  if (displayWidth > maxV1WidthFactor) {
    displayWidth = maxV1WidthFactor
  }
  const computedWidth = displayWidth * v1MultiplicationFactor
  // should be more than v2 max width 400px
  return (computedWidth < 400 ? computedWidth : 400) as number
}

export const getPassageContent = (data) => {
  let rawPassageContent = ''
  let passageContent = ''
  const delimiter = ' '
  if (!_.isEmpty(data)) {
    data.forEach((item) => {
      if (item.title) {
        rawPassageContent += delimiter + item.title
      }
      if (item.heading) {
        rawPassageContent += delimiter + item.heading
      }
      if (item.content) {
        rawPassageContent += delimiter + item.content
      }
    })
    passageContent = htmlToPlainText(
      sanitizeHtml(rawPassageContent, {
        allowedTags: [],
        allowedAttributes: {},
      })
    )
  }
  return passageContent
}

export const createChangeStream = (collection) => {
  Logger.info(`Creating stream watch for ${collection}`)
  const dbCollection: any = models[collection]
  const changeStream = dbCollection.watch()
  changeStream.on('change', (data) => {
    if (data.fullDocument && data.fullDocument.v1Id) {
      Logger.info(`${collection} created with v1Id: ${data.fullDocument.v1Id}`)
    } else {
      Logger.info(`${collection} updated _id: ${data.documentKey._id}`)
    }
  })
  return changeStream
}

  if (changeStream) {
    Logger.info(`Closing stream watch for ${collection}`)
    return changeStream.close()
  }
}
