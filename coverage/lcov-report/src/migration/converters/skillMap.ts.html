export const skillMapConverter = async (rows): Promise<compactAlignment[]> => {
  const standardModel = new StandardModel()
  const v1SkillIds = rows.reduce((n, o) => {
    n.push(o.master_skill_id)
    return n
  }, [])
 
  // we don't need to load because standardIds are id,
  // curriculumIds are taxonomy_id in the above query ??
  const v2StandardRows = await standardModel.Standard.find({
    _id: { $in: v1SkillIds },
    level: { $in: ['ELO', 'SUB_ELO'] },
  })
 
  const returnObj = uniqBy(v2StandardRows, '_id').reduce((acc, cur) => {
    const { _id, curriculumId } = cur
    if (!acc.find((curr) => curr.curriculumId === curriculumId)) {
      acc.push({
        curriculumId,
        standardIds: [_id],
      })
      return acc
    }
    acc.forEach((alignment) => {
      if (alignment.curriculumId === curriculumId) {
        alignment.standardIds.push(_id)
      }
    })
    return acc
  }, [] as compactAlignment[])
 
  return returnObj
}
 
export const skillMap = async (v1Id, questionSkills?) => {
  let rows
  if (!questionSkills) {
    rows = await knex
      .select('master_skill_id', 'question_id')
      .table('t_question_skill_map')
      .where({ question_id: v1Id })
  } else {
    rows = questionSkills
  }
  if (!rows.length) {
    console.log('question Skill Map not found')
    return []
  }
  return skillMapConverter(rows)
}
