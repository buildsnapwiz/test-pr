import knex from '../../services/mysql'
import commons from '../../utils/commons'
import { proficiencyMethodMap } from './const'
import StandardsProficiencyModel from '../../models/standardsProficiencySettings'
import utils from './utils'
import Logger from '../../logger'
 
const defaultMasteryScale = require('../../controllers/masteryScale.json')
 
const options = { upsert: true, new: true, timestamps: false }
 
const v1ColorToV2ColorMap = {
  '#c8eb9b': '#3DB04E',
  '#f3fccf': '#74E27A',
  '#fafa2b': '#AFA515',
  '#fdfdc8': '#EBDD54',
  '#ffc300': '#F39300',
  '#fde2b3': '#FEC571',
  '#fabdbd': '#576BA9',
  '#d4e9fa': '#A1C3EA',
}
 
const transformStandardsProficiency = (rows, v1DistrictId, v2DistrictId) => {
  const scale = rows.map((row) => {
    return {
      v1Id: row.id,
      score: row.score,
      masteryLevel: row.name,
      shortName: row.short_name,
      threshold: row.threshold,
      color: v1ColorToV2ColorMap[`${row.color_code}`.toLowerCase()],
    }
  })
  const calcType = proficiencyMethodMap[rows[0].calc_method]
  const calcAttribute = rows[0].calc_attr
  let v1Attributes: any = {
    v1CreatedBy: rows[0].created_by,
    v1ModifiedBy: rows[0].modified_by,
    v1Status: rows[0].status,
  }
  v1Attributes = utils.removeNullOrUndefinedFieldsWithKey(v1Attributes)
  const spObject = {
    name: 'Standards Mastery', // V1 doesn't have this
    v1OrgId: v1DistrictId,
    orgType: 'district',
    orgId: v2DistrictId,
    updatedAt: commons.getTimeStamp(rows[0].modified_date),
    createdAt: commons.getTimeStamp(rows[0].created_date),
    scale,
    calcType,
    calcAttribute,
  }
 
  Object.assign(spObject, {
    v1Attributes,
  })
 
  if (calcType === 'DECAYING_AVERAGE') {
    Object.assign(spObject, {
      decay: calcAttribute,
    })
  }
  if (calcType === 'MOVING_AVERAGE') {
    Object.assign(spObject, {
      noOfAssessments: calcAttribute,
    })
  }
  return spObject
}
 
export const migrateStandardsProficiency = async (
  v1DistrictId,
  v2DistrictId,
  archiveMigration = false,
  archivedConnection = null
) => {
  let rows = await knex
    .select()
    .table('t_mastery_grading_scale')
    .where({ district_id: v1DistrictId })
  let standardsProficiency = {}
  if (rows.length) {
    standardsProficiency = transformStandardsProficiency(
      rows,
      v1DistrictId,
      v2DistrictId
    )
  } else {
    standardsProficiency = {
      orgType: 'district',
      v1OrgId: v1DistrictId,
      orgId: v2DistrictId,
      calcType: 'DECAYING_AVERAGE',
      calcAttribute: 60,
      name: 'Standard Mastery',
      decay: 60,
      scale: defaultMasteryScale,
      noOfAssessments: 5,
      institutionIds: [],
    }
  }
 
  let standardsProficiencies = await new StandardsProficiencyModel().Settings.findOne(
    {
      v1OrgId: v1DistrictId,
      updatedAt: { $gt: standardsProficiency['updatedAt'] || 0 },
    }
  ).lean()
  if (!standardsProficiencies) {
    try {
      standardsProficiencies = await new StandardsProficiencyModel(
        archiveMigration ? archivedConnection : null
      ).Settings.findOneAndUpdate(
        { v1OrgId: v1DistrictId },
        standardsProficiency,
        options
      ).lean()
    } catch (duex) {
      if (duex.message.includes('duplicate key error')) {
        Logger.info(
          `Standards Proficiency duplicate encountered for v1Id ${v1DistrictId}`
        )
        standardsProficiencies = await new StandardsProficiencyModel(
          archiveMigration ? archivedConnection : null
        ).Settings.findOne({
          v1OrgId: v1DistrictId,
        }).lean()
      } else {
        throw duex
      }
    }
  }
  return standardsProficiencies
}
