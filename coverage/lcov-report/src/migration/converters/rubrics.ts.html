import _ from 'lodash'
import RubricsModel from '../../rubrics/rubricsModel'
import UserModel from '../../models/user'
import DistrictModel from '../../models/district'
import knex from '../../services/mysql'
import commons from '../../utils/commons'
import { RubricStatusAndActiveFlagMap } from './const'
import Logger from '../../logger'
 
const options = { upsert: true, new: true, timestamps: false }
 
const rubricsModel = new RubricsModel()
const userModel = new UserModel()
const districtModel = new DistrictModel()
 
const getUserDetail = (user) => {
  if (user) {
    return {
      name: _.without(
        [user.firstName, user.lastName],
        undefined,
        null,
        ''
      ).join(' '),
      _id: user._id,
    }
  }
  return {}
}
 
const transformRubrics = (
  rubric,
  v2DistrictId,
  createdByUser,
  modifiedByUser,
  sharedWithDistrict
) => {
  return {
    v1Id: rubric.id,
    districtId: v2DistrictId,
    name: rubric.name,
    description: rubric.description,
    createdBy: { ...getUserDetail(createdByUser) },
    modifiedBy: { ...getUserDetail(modifiedByUser) },
    status: RubricStatusAndActiveFlagMap[rubric.status].status,
    active: RubricStatusAndActiveFlagMap[rubric.status].active,
    criteria: _.get(
      JSON.parse(_.get(rubric, 'criteria_json', '{}')),
      'criterias',
      []
    ).map((criteria) => {
      criteria.id = criteria.UUID
      delete criteria.UUID
      criteria.ratings = _.get(criteria, 'ratings', []).map((rating) => {
        rating.id = rating.UUID
        delete rating.UUID
        return rating
      })
      return criteria
    }),
    createdAt: commons.getTimeStamp(rubric.created_date),
    updatedAt: commons.getTimeStamp(rubric.modified_date),
    sharedWith: sharedWithDistrict,
  }
}
 
export const migrateRubrics = async (
  v1RubricsId,
  v1DistrictId,
  archiveMigration = false,
  archivedConnection = null
) => {
  let filter = {}
  if (_.isNumber(v1RubricsId)) {
    console.log('migrating rubrcis by id', v1RubricsId)
    filter = { id: v1RubricsId }
  } else {
    console.log('migrating rubrcis by district id', v1DistrictId)
    filter = { district_id: v1DistrictId }
  }
  const rubrics = await knex.select().table('t_rubric_details').where(filter)
 
  if (!rubrics.length) {
    throw new Error(
      `No rubrcis found for id/districtId ${v1RubricsId / v1DistrictId}`
    )
  }
  const documents: any[] = []
  let rubrc: any
  for (const rubric of rubrics) {
    const { modified_by, created_by } = rubric
    let createdByUser = await userModel.User.findOne(
      { v1Id: created_by },
      { firstName: 1, lastName: 1 }
    )
    if (archiveMigration && !createdByUser) {
      createdByUser = await new UserModel(archivedConnection).User.findOne(
        { v1Id: created_by },
        { firstName: 1, lastName: 1 }
      )
    }
    let modifiedByUser
    if (modified_by === created_by) {
      modifiedByUser = createdByUser
    } else {
      modifiedByUser = await userModel.User.findOne(
        { v1Id: modified_by },
        { firstName: 1, lastName: 1 }
      )
      if (archiveMigration && !modifiedByUser) {
        modifiedByUser = await new UserModel(archivedConnection).User.findOne(
          { v1Id: modified_by },
          { firstName: 1, lastName: 1 }
        )
      }
    }
    let district = {}
    if (_.isNumber(rubric.district_id)) {
      // @ts-ignore
      district = await districtModel.District.findOne(
        { v1Id: rubric.district_id },
        { _id: 1 }
      ).lean()
      if (archiveMigration && !district) {
        district = await new DistrictModel(archivedConnection).District.findOne(
          { v1Id: rubric.district_id },
          { _id: 1 }
        ).lean()
      }
    }
 
    const sharedWithDistrict: any = []
    if (
      !_.isNil(rubric.shared_with_ids) &&
      rubric.shared_with_ids.includes('d_')
    ) {
      const dId = rubric.shared_with_ids.split('_')[1]
      let sharedDistrict = await districtModel.District.findOne(
        { v1Id: dId },
        { _id: 1, name: 1 }
      ).lean()
      if (archiveMigration && !sharedDistrict) {
        sharedDistrict = await new DistrictModel(
          archivedConnection
        ).District.findOne({ v1Id: dId }, { _id: 1, name: 1 }).lean()
      }
      if (sharedDistrict) {
        sharedWithDistrict.push({
          _id: sharedDistrict._id,
          name: sharedDistrict.name,
        })
      }
    }
 
    const rubricDoc = transformRubrics(
      rubric,
      _.get(district, '_id'),
      createdByUser,
      modifiedByUser,
      sharedWithDistrict
    )
    rubrc = await rubricsModel.Rubrics.findOne({
      v1Id: rubric.id,
      updatedAt: { $gt: rubricDoc.updatedAt || 0 },
    })
    if (archiveMigration && !rubrc) {
      rubrc = await new RubricsModel(archivedConnection).Rubrics.findOne({
        v1Id: rubric.id,
        updatedAt: { $gt: rubricDoc.updatedAt || 0 },
      })
    }
    if (!rubrc) {
      documents.push({
        updateMany: {
          filter: {
            v1Id: rubric.id,
          },
          update: rubricDoc,
          ...options,
        },
      })
    }
  }
  if (!documents.length && rubrc) {
    return rubrc
  }
  if (documents.length > 1) {
    const result = await new RubricsModel(
      archiveMigration ? archivedConnection : null
    ).Rubrics.bulkWrite(documents)
    return result
  }
  try {
    return await new RubricsModel(
      archiveMigration ? archivedConnection : null
    ).Rubrics.findOneAndUpdate(
      documents[0].updateMany.filter,
      documents[0].updateMany.update,
      options
    )
  } catch (duex) {
    if (duex.message.includes('duplicate key error')) {
      Logger.info(
        `Rubrics duplicate encountered for v1Id ${JSON.stringify(filter)}`
      )
      return await new RubricsModel(
        archiveMigration ? archivedConnection : null
      ).Rubrics.findOne(documents[0].updateMany.filter)
    }
    throw duex
  }
}
